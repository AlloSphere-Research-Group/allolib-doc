546 inline bool aeq(float a, float b, int maxULP) {
547  // Make sure maxULP is non-negative and small enough that the
548  // default NAN won't compare as equal to anything.
549  // assert(maxULP > 0 && maxULP < 4 * 1024 * 1024);
550  union {
551  float f;
552  int32_t i;
553  } u;
554  u.f = a;
555  int32_t ai = u.i;
556  u.f = b;
557  int32_t bi = u.i;
558  // Make ai and bi lexicographically ordered as a twos-complement int
559  if (ai < 0) ai = 0x80000000 - ai;
560  if (bi < 0) bi = 0x80000000 - bi;
561  return std::abs((float)(ai - bi)) <= maxULP;
562 }

564 inline bool aeq(double a, double b, int maxULP) {
565  // Make sure maxULP is non-negative and small enough that the
566  // default NAN won't compare as equal to anything.
567  // assert(maxULP > 0 && maxULP < 4 * 1024 * 1024);
568  union {
569  double f;
570  int64_t i;
571  } u;
572  u.f = a;
573  int64_t ai = u.i;
574  u.f = b;
575  int64_t bi = u.i;
576  // Make ai and bi lexicographically ordered as a twos-complement int
577  if (ai < 0) ai = 0x8000000000000000ULL - ai;
578  if (bi < 0) bi = 0x8000000000000000ULL - bi;
579  return std::abs((double)(ai - bi)) <= maxULP;
580 }

586 TEM T atan2Fast(const T& y, const T& x) {
587  T r, angle;
588  T ay = std::abs(y) + T(1e-10); // kludge to prevent 0/0 condition
589 
590  if (x < T(0)) {
591  r = (x + ay) / (ay - x);
592  angle = T(M_3PI_4);
593  } else {
594  r = (x - ay) / (x + ay);
595  angle = T(M_PI_4);
596  }
597 
598  angle += (T(0.1963) * r * r - T(0.9817)) * r;
599  return y < T(0) ? -angle : angle;
600 }

703  // T res = 1, bin = 1;
704  //

724 TEM T legendreP(int l, int m, T ct, T st) {
725  switch (l) {
726  case 0:
727  return 1.;
728 
729  case 1:
730  switch (m) {
731  case 0:
732  return ct;
733  case 1:
734  return -st;
735  default:
736  return 0.;
737  }
738 
739  case 2:
740  switch (m) {
741  case 0:
742  return -0.5 + 1.5 * ct * ct;
743  case 1:
744  return -3.0 * ct * st;
745  case 2:
746  return 3.0 * st * st;
747  default:
748  return 0.;
749  }
750 
751  case 3:
752  switch (m) {
753  case 0:
754  return ct * (-1.5 + 2.5 * ct * ct);
755  case 1:
756  return (1.5 - 7.5 * ct * ct) * st;
757  case 2:
758  return 15. * ct * st * st;
759  case 3:
760  return -15. * st * st * st;
761  default:
762  return 0.;
763  }
764 
765  case 4:
766  switch (m) {
767  case 0:
768  ct *= ct;
769  return 0.375 + ct * (-3.75 + 4.375 * ct);
770  case 1:
771  return ct * (7.5 - 17.5 * ct * ct) * st;
772  case 2:
773  return (-7.5 + 52.5 * ct * ct) * st * st;
774  case 3:
775  return -105. * ct * st * st * st;
776  case 4:
777  st *= st;
778  return 105. * st * st;
779  default:
780  return 0.;
781  }
782 
783  default:;
784  }
785 
786  // if(l<0){ /*printf("l=%d. l must be non-negative.\n");*/ return 0; }
787  // if(m<-l || m>l){ /*printf("m=%d. m must be -l <= m <= l.\n");*/ return 0;
788  // }
789 
790  // First compute answer for |m|
791 
792  // compute P_l^m(x) by the recurrence relation
793  // (l-m)P_l^m(x) = x(2l-1)P_{l-1}^m(x) - (l+m-1)P_{l-2}^m(x)
794  // with
795  // P_m^m(x) = (-1)^m (2m-1)!! (1-x)^{m/2},
796  // P_{m+1}^m(x) = x(2m+1) P_m^m(x).
797 
798  T P = 0; // the result
799  int M = std::abs((double)m); // M = |m|
800  T y1 = 1.; // recursion state variable
801 
802  for (int i = 1; i <= M; ++i) y1 *= -((i << 1) - 1) * st;
803 
804  if (l == M)
805  P = y1;
806 
807  else {
808  T y = ((M << 1) + 1) * ct * y1;
809  if (l == (M + 1))
810  P = y;
811 
812  else {
813  T c = (M << 1) - 1;
814  for (int k = M + 2; k <= l; ++k) {
815  T y2 = y1;
816  y1 = y;
817  T d = c / (k - M);
818  y = (2. + d) * ct * y1 - (1. + d) * y2;
819  }
820  P = y;
821  }
822  }
823 
824  // // In the case that m<0,
825  // // compute P_n^{-|m|}(x) by the formula
826  // // P_l^{-|m|}(x) = (-1)^{|m|}((l-|m|)!/(l+|m|)!)^{1/2} P_l^{|m|}(x).
827  // // NOTE: when l and |m| are large, we risk numerical underflow...
828  // if(m<0){
829  // for(int i=l-M+1; i<=l+M; ++i) P *= 1. / i;
830  // if(al::odd(M)) P = -P;
831  // }
832 
833  return P;
834 }
