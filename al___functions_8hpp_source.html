461 inline bool aeq(float a, float b, int maxULP){
462  // Make sure maxULP is non-negative and small enough that the
463  // default NAN won't compare as equal to anything.
464  //assert(maxULP > 0 && maxULP < 4 * 1024 * 1024);
465  union{ float f; int32_t i; } u;
466  u.f=a; int32_t ai = u.i;
467  u.f=b; int32_t bi = u.i;
468  // Make ai and bi lexicographically ordered as a twos-complement int
469  if(ai < 0) ai = 0x80000000 - ai;
470  if(bi < 0) bi = 0x80000000 - bi;
471  return std::abs((float)(ai - bi)) <= maxULP;
472 }

489 TEM T atan2Fast(const T& y, const T& x){
490 
491  T r, angle;
492  T ay = std::abs(y) + T(1e-10); // kludge to prevent 0/0 condition
493 
494  if(x < T(0)){
495  r = (x + ay) / (ay - x);
496  angle = T(M_3PI_4);
497  }
498  else{
499  r = (x - ay) / (x + ay);
500  angle = T(M_PI_4);
501  }
502 
503  angle += (T(0.1963)*r*r - T(0.9817))*r;
504  return y < T(0) ? -angle : angle;
505 }

620 TEM T legendreP(int l, int m, T ct, T st){
621 
622  switch(l){
623  case 0: return 1.;
624 
625  case 1:
626  switch(m){
627  case 0: return ct;
628  case 1: return -st;
629  default:return 0.;
630  }
631 
632  case 2:
633  switch(m){
634  case 0: return -0.5 + 1.5*ct*ct;
635  case 1: return -3.0*ct*st;
636  case 2: return 3.0*st*st;
637  default:return 0.;
638  }
639 
640  case 3:
641  switch(m){
642  case 0: return ct*(-1.5 + 2.5*ct*ct);
643  case 1: return (1.5 - 7.5*ct*ct)*st;
644  case 2: return 15.*ct*st*st;
645  case 3: return -15.*st*st*st;
646  default:return 0.;
647  }
648 
649  case 4:
650  switch(m){
651  case 0: ct*=ct; return 0.375 + ct*(-3.75 + 4.375*ct);
652  case 1: return ct*(7.5 - 17.5*ct*ct)*st;
653  case 2: return (-7.5 + 52.5*ct*ct)*st*st;
654  case 3: return -105.*ct*st*st*st;
655  case 4: st*=st; return 105.*st*st;
656  default:return 0.;
657  }
658 
659  default:;
660  }
661 
662 // if(l<0){ /*printf("l=%d. l must be non-negative.\n");*/ return 0; }
663 // if(m<-l || m>l){ /*printf("m=%d. m must be -l <= m <= l.\n");*/ return 0; }
664 
665  // First compute answer for |m|
666 
667  // compute P_l^m(x) by the recurrence relation
668  // (l-m)P_l^m(x) = x(2l-1)P_{l-1}^m(x) - (l+m-1)P_{l-2}^m(x)
669  // with
670  // P_m^m(x) = (-1)^m (2m-1)!! (1-x)^{m/2},
671  // P_{m+1}^m(x) = x(2m+1) P_m^m(x).
672 
673  T P = 0; // the result
674  int M = std::abs((double)m); // M = |m|
675  T y1 = 1.; // recursion state variable
676 
677  for(int i=1; i<=M; ++i)
678  y1 *= -((i<<1) - 1) * st;
679 
680  if(l==M) P = y1;
681 
682  else{
683  T y = ((M<<1) + 1) * ct * y1;
684  if(l==(M+1)) P = y;
685 
686  else{
687  T c = (M<<1) - 1;
688  for(int k=M+2; k<=l; ++k){
689  T y2 = y1;
690  y1 = y;
691  T d = c / (k - M);
692  y = (2. + d) * ct * y1 - (1. + d) * y2;
693  }
694  P = y;
695  }
696  }
697 
698 // // In the case that m<0,
699 // // compute P_n^{-|m|}(x) by the formula
700 // // P_l^{-|m|}(x) = (-1)^{|m|}((l-|m|)!/(l+|m|)!)^{1/2} P_l^{|m|}(x).
701 // // NOTE: when l and |m| are large, we risk numerical underflow...
702 // if(m<0){
703 // for(int i=l-M+1; i<=l+M; ++i) P *= 1. / i;
704 // if(al::odd(M)) P = -P;
705 // }
706 
707  return P;
708 }
