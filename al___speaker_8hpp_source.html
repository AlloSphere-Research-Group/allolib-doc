81  template <class T>
82  Speaker& posCart(T* xyz) {
83  using namespace std;
84  float elr = toRad(elevation);
85  float azr = toRad(azimuth);
86  float cosel = cos(elr);
87  xyz[0] = cos(azr) * cosel * radius;
88  xyz[1] = sin(azr) * cosel * radius;
89  xyz[2] = sin(elr) * radius;
90  return *this;
91  }

93  void posCart2(Vec3d xyz) {
94  using namespace std;
95 
96  radius =
97  sqrt(float((xyz[0] * xyz[0]) + (xyz[1] * xyz[1]) + (xyz[2] * xyz[2])));
98  float gd = sqrt(float((xyz[0] * xyz[0]) + (xyz[1] * xyz[1])));
99  elevation = atan2f(float(xyz[2]), gd) * 180.f / float(M_PI);
100  azimuth = atan2f(float(xyz[0]), float(xyz[1])) * 180.f / float(M_PI);
101  }

104  Vec3d vec() const {
105  // TODO doxygen style commenting on coordinates like ambisonics
106  double cosel = cos(toRad(double(elevation)));
107  double x = cos(toRad(double(azimuth))) * cosel * double(radius);
108  double y = sin(toRad(double(azimuth))) * cosel * double(radius);
109  double z = sin(toRad(double(elevation))) * double(radius);
110  // Ryan: the standard conversions assume +z is up, these are correct for
111  // allocore

113  // double x = sin(toRad(azimuth)) * cosel * radius;
114  // double y = sin(toRad(elevation)) * radius;
115  // double z = -1*cos(toRad(azimuth)) * cosel * radius;
116  return Vec3d(x, y, z);
