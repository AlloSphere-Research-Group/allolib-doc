286 PRE void CLS::fade(int fadeOutFrames, int fadeInFrames){
287  if(fadeInFrames > 0){
288  double amp;
289  double slope = 1./(fadeInFrames-1);
290  for(int c=0; c<channels(); ++c){
291  amp = 0.;
292  for(int i=0; i<fadeInFrames; ++i){
293  sample(i,c) *= amp;
294  amp += slope;

299  if(fadeOutFrames > 0){
300  double amp;
301  double slope =-1./(fadeOutFrames-1);
302  for(int c=0; c<channels(); ++c){
303  amp = 1;
304  for(int i=frames()-1-fadeOutFrames; i<frames(); ++i){
305  sample(i,c) *= amp;
306  amp += slope;
307  }
308  }
309  }
