247  Tf d2 = d * d;

282 template <class Tf, class Tv>
283 inline void cardinalSpline(Tv * w, const Tv * x, const Tf& f, double b){
284 
285  //c = (1-b); // tension
286  b *= (x[2]-x[1]); // make domain [t[1], t[2]]
287 
288  // evaluate the Hermite basis functions
289  Tf h00 = (1 + 2*f)*(f-1)*(f-1);
290  Tf h10 = f*(f-1)*(f-1);
291  Tf h01 = f*f*(3-2*f);
292  Tf h11 = f*f*(f-1);
293 
294  /*
295  The general cubic Hermite spline is
296  p(f) = h00 * p0 + h10 * m0 + h01 * p1 + h11 * m1
297 
298  For a cardinal spline, the tangent at point k is
299  m[k] = (1-c) * (p[k+1] - p[k-1]) / (x[k+1] - x[k-1])
300 
301  To get the weights for each point, we plug the tangents into the general
302  spline equation and factor out the p[k].
303  */
304 
305  w[0] = ( - b*h10/(x[2]-x[0]));
306  w[1] = (h00 - b*h11/(x[3]-x[1]));
307  w[2] = (h01 + b*h10/(x[2]-x[0]));
308  w[3] = ( + b*h11/(x[3]-x[1]));
309 }

312 inline Tv cubic(Tf f, const Tv& w, const Tv& x, const Tv& y, const Tv& z){
313 // Tv c3 = (x - y)*(Tf)1.5 + (z - w)*(Tf)0.5;
314 // Tv c2 = w - x*(Tf)2.5 + y*(Tf)2. - z*(Tf)0.5;
315 // Tv c1 = (y - w)*(Tf)0.5;
316 // return ((c3 * f + c2) * f + c1) * f + x;
317 
318  // -w + 3x - 3y + z
319  // 2w - 5x + 4y - z
320  // c2 = w - 2x + y - c3
321 
322 // Tv c3 = (x - y)*(Tf)3 + z - w;
323 // Tv c2 = w - x*(Tf)2 + y - c3;
324 // Tv c1 = y - w;
325 // return (((c3 * f + c2) * f + c1)) * f * (Tf)0.5 + x;
326 
327 // Tv c3 = (x - y)*(Tf)1.5 + (z - w)*(Tf)0.5;
328 // Tv c2 = (y + w)*(Tf)0.5 - x - c3;
329 // Tv c1 = (y - w)*(Tf)0.5;
330 // return ((c3 * f + c2) * f + c1) * f + x;
331 
332  Tv c1 = (y - w)*Tf(0.5);
333  Tv c3 = (x - y)*Tf(1.5) + (z - w)*Tf(0.5);
334  Tv c2 = c1 + w - x - c3;
335  return ((c3 * f + c2) * f + c1) * f + x;

360 template <class Tp, class Tv>
361 inline Tv hermite(Tp f,
362  const Tv& w, const Tv& x, const Tv& y, const Tv& z,
363  Tp tension, Tp bias)
364 {
365  tension = (Tp(1) - tension)*Tp(0.5);
366 
367  // compute endpoint tangents
368  //Tv m0 = ((x-w)*(1+bias) + (y-x)*(1-bias))*tension;
369  //Tv m1 = ((y-x)*(1+bias) + (z-y)*(1-bias))*tension;
370  Tv m0 = ((x*Tv(2) - w - y)*bias + y - w)*tension;
371  Tv m1 = ((y*Tv(2) - x - z)*bias + z - x)*tension;
372 
373 // x - w + x b - w b + y - x - y b + x b
374 // -w + 2x b - w b + y - y b
375 // b(2x - w - y) + y - w
376 //
377 // y - x + y b - x b + z - y - z b + y b
378 // -x + 2y b - x b + z - z b
379 // b(2y - x - z) + z - x
380 
381  Tp f2 = f * f;
382  Tp f3 = f2 * f;
383 
384  // compute hermite basis functions
385  Tp a3 = Tp(-2)*f3 + Tp(3)*f2;
386  Tp a0 = Tp(1) - a3;
387  Tp a2 = f3 - f2;
388  Tp a1 = f3 - Tp(2)*f2 + f;
389 
390  return x*a0 + m0*a1 + m1*a2 + y*a3;
391 }

393 template <class T> void lagrange(T * a, T delay, int order){
394  for(int i=0; i<=order; ++i){
395  T coef = T(1);
396  T i_f = T(i);
397  for(int j=0; j<=order; ++j){
398  if(j != i){
399  T j_f = (T)j;
400  coef *= (delay - j_f) / (i_f - j_f);
401  }
402  }
403  *a++ = coef;
404  }
405 }
