554  slope = cov(x, y) / var(x)
555  inter = x_mean - slope * y_mean

557 template <class T, class T2, class T3>
558 void lineFit(const T * src, unsigned len, T2& slope, T3& inter){
559  T lenT = T(len);
560  T meanX = (lenT - T(1)) * T(0.5); // mean of independent variables (the indices)
561  T meanY = sum(src, len) / lenT; // mean of dependent variables
562  T sumSqrs = meanX*(meanX+T(1))*(meanX*T(2./6) + T(1./6));
563  T varX = T(2)*sumSqrs; // variance of x
564 
565  T cov = T(0); // covariance
566  T dx =-meanX; // current distance from point to center along x
567 
568  LOOP(len, 1){
569  cov += dx++ * (*src++ - meanY);
570  }
571 
572  slope = cov / varX;
573  inter = meanY - slope * meanX;
574 }
