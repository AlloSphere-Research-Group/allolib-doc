133 inline void OmniRenderer::draw_using_perprojection_capture() {
134  // start drawing to perprojection fbos
135 
136  mGraphics.omni(true); // set true to use omni default shaders when drawing
137  // lens and pose for rendering is set in PerProjectionRender::begin
138  // so updating those in onDraw will not have effect in rendering
139  // will setting up omni rendering,
140  // begin also pushes fbo, viewport, viewmat, projmat, lens, shader
141  pp_render.begin(mGraphics, lens(), pose());
142  glDrawBuffer(GL_COLOR_ATTACHMENT0); // for fbo's output
143  if (render_stereo) {
144  for (int eye = 0; eye < 2; eye += 1) {
145  pp_render.set_eye(eye);
146  for (int i = 0; i < pp_render.num_projections(); i++) {
147  pp_render.set_projection(i);
148  mGraphics.depthTesting(true);
149  mGraphics.depthMask(true);
150  mGraphics.blending(false);
151  onDraw(mGraphics);
152  }
153  }
154  } else {
155  // std::cout << "rendering eye " << eye_to_render << std::endl;
156  pp_render.set_eye(eye_to_render);
157  for (int i = 0; i < pp_render.num_projections(); i++) {
158  pp_render.set_projection(i);
159  mGraphics.depthTesting(true);
160  mGraphics.depthMask(true);
161  mGraphics.blending(false);
162  onDraw(mGraphics);
163  }
164  }
165  pp_render.end(); // pops everything pushed before
166 
167  /* Settings for warp and blend composition sampling
168  */
169  mGraphics.omni(false); // warp and blend composition done on flat rendering
170  mGraphics.eye(Graphics::MONO_EYE); // stereo handled at capture stage
171  mGraphics.polygonMode(Graphics::FILL); // to draw viewport filling quad
172  mGraphics.blending(false); // blending already done when capturing
173  mGraphics.depthTesting(false); // no depth testing when drawing viewport slab
174  mGraphics.pushViewport(0, 0, fbWidth(),
175  fbHeight()); // filling the whole window
176 
177  // now sample the results
178  if (running_in_sphere_renderer) {
179  if (window_is_stereo_buffered) {
180  // rendering stereo in sphere
181  glDrawBuffer(GL_BACK_LEFT);
182  mGraphics.clearColor(0, 0, 0);
183  mGraphics.clearDepth(1);
184  pp_render.composite(mGraphics, 0);
185  glDrawBuffer(GL_BACK_RIGHT);
186  mGraphics.clearColor(0, 0, 0);
187  mGraphics.clearDepth(1);
188  pp_render.composite(mGraphics, 1);
189  } else { // rendering mono in sphere
190  // std::cout << "sampling mono in sphere setup" << std::endl;
191  glDrawBuffer(GL_BACK_LEFT);
192  mGraphics.clearColor(1, 0, 0);
193  mGraphics.clearDepth(1);
194  pp_render.composite(mGraphics, (eye_to_render == 1) ? 1 : 0);
195  }
196  } else {
197  if (window_is_stereo_buffered) {
198  // rendering stereo on display other than sphere
199  glDrawBuffer(GL_BACK_LEFT);
200  mGraphics.clearColor(0, 0, 0);
201  mGraphics.clearDepth(1);
202  pp_render.composite_desktop(mGraphics, 0); // texture[0]: left
203  glDrawBuffer(GL_BACK_RIGHT);
204  mGraphics.clearColor(0, 0, 0);
205  mGraphics.clearDepth(1);
206  pp_render.composite_desktop(mGraphics, 1); // texture[1]: right
207  } else { // rendering mono on display other than sphere
208  // std::cout << "sampling mono on flat display" << std::endl;
209  glDrawBuffer(GL_BACK_LEFT);
210  mGraphics.clearColor(0.2, 0.2, 0.2);
211  mGraphics.clearDepth(1);
212  pp_render.composite_desktop(
213  mGraphics,
214  (eye_to_render == 1) ? 1 : 0 // mono and left eye is
215  // rendered on texture[0],
216  // right eye is on texture[1]
217  );
218  }
219  }
220  mGraphics.popViewport();
221  // put back default drawbuffer
222  glDrawBuffer(GL_BACK_LEFT);
223 }
