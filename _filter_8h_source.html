752 template <class Tv, class Tp, class Td>

754  mLevel=v;
755  switch(mType){
756  case PEAKING:

758  break;
759  case LOW_SHELF:
760  case HIGH_SHELF:

769 template <class Tv, class Tp, class Td>
770 inline void Biquad<Tv,Tp,Td>::resRecip(Tp v){
771  mResRecip = v;
772  mAlpha = mImag * mResRecip * mBeta;
773 
774  switch(mType){
775  case LOW_SHELF: case HIGH_SHELF: break; // coefs computed in type()
776  default:
777  // Note: b_0 is assumed to be equal to 1 in the difference equation.
778  // For this reason, we divide all other coefficients by b_0.
779  mB[0] = Tp(1) / (Tp(1) + mAlpha); // 1/b_0
780  mB[1] = Tp(-2) * mReal * mB[0];
781  mB[2] = (Tp(1) - mAlpha) * mB[0];
782  }
783 
784  type(mType);
785 }

792 template <class Tv, class Tp, class Td>
793 inline void Biquad<Tv,Tp,Td>::type(FilterType typeA){
794  mType = typeA;
795 
796  switch(mType){
797  case LOW_PASS:
798  mA[1] = (Tp(1) - mReal) * mB[0];
799  mA[0] = mA[1] * Tp(0.5);
800  mA[2] = mA[0];
801  break;
802  case HIGH_PASS: // low-pass with odd k a_k and freq flipped
803  mA[1] = (Tp(-1) - mReal) * mB[0];
804  mA[0] = mA[1] * Tp(-0.5);
805  mA[2] = mA[0];
806  break;
807  case RESONANT:
808  mA[0] = mImag * Tp(0.5) * mB[0];
809  mA[1] = Tp(0);
810  mA[2] =-mA[0];
811  break;
812  case BAND_PASS:
813  mA[0] = mAlpha * mB[0];
814  mA[1] = Tp(0);
815  mA[2] =-mA[0];
816  break;
817  case BAND_REJECT:
818  mA[0] = mB[0];
819  mA[1] = mB[1];
820  mA[2] = mB[0];
821  break;
822  case ALL_PASS:
823  mA[0] = mB[2];
824  mA[1] = mB[1];
825  mA[2] = Tp(1);
826  break;
827  case PEAKING:{
828  Tp alpha_A_b0 = mAlpha * mLevel * mB[0];
829  mA[0] = mB[0] + alpha_A_b0;
830  mA[1] = mB[1];
831  mA[2] = mB[0] - alpha_A_b0;
832  }
833  break;
834  case LOW_SHELF:{
835  Tp A = mBeta*mBeta*Tp(0.25); // sqrt(mLevel)
836  Tp Ap1 = A + Tp(1), Am1 = A - Tp(1);
837  mB[0] = Tp(1)/(Ap1 + Am1*mReal + mAlpha); // 1/b_0
838  mB[1] = Tp(-2)*(Am1 + Ap1*mReal ) * mB[0];
839  mB[2] = (Ap1 + Am1*mReal - mAlpha) * mB[0];
840  mA[0] = A*(Ap1 - Am1*mReal + mAlpha) * mB[0];
841  mA[1] = Tp( 2)*A*(Am1 - Ap1*mReal ) * mB[0];
842  mA[2] = A*(Ap1 - Am1*mReal - mAlpha) * mB[0];
843  }
844  break;
845  case HIGH_SHELF:{
846  Tp A = mBeta*mBeta*Tp(0.25); // sqrt(mLevel)
847  Tp Ap1 = A + Tp(1), Am1 = A - Tp(1);
848  mB[0] = Tp(1)/(Ap1 - Am1*mReal + mAlpha); // 1/b_0
849  mB[1] = Tp(2)*(Am1 - Ap1*mReal ) * mB[0];
850  mB[2] = (Ap1 - Am1*mReal - mAlpha) * mB[0];
851  mA[0] = A*(Ap1 + Am1*mReal + mAlpha) * mB[0];
852  mA[1] =Tp(-2)*A*(Am1 + Ap1*mReal ) * mB[0];
853  mA[2] = A*(Ap1 + Am1*mReal - mAlpha) * mB[0];
854  }
855  break;
856  default:;
857  };
858 }

906 inline void OnePole<Tv,Tp,Td>::freq(Tp v){
907  mFreq = v;
908 
909  // |H(w)| = |a0| / sqrt(1 + b1^2 + 2 b1 cos w)
910 
911  switch(mType){
912  default:{ // low-pass
913  // cutoff based on pole at DC (inaccurate with large bandwidth)
914  //mB1 = poleRadius(Tp(2) * v * Td::ups());
915  // b1 found by setting |H(w)| = 0.707
916  Tp re = getReal(v * Td::ups());
917  Tp p1 = re - Tp(2);
918  mB1 = -(p1 + sqrt(p1*p1 - Tp(1)));
919  mA0 = Tp(1) - mB1;
920  }
921  break;
922  case HIGH_PASS:{
923  // cutoff based on pole at Nyquist (inaccurate with large bandwidth)
924  //mB1 = -poleRadius(Tp(1) - Tp(2) * v * Td::ups());
925  // b1 found by setting |H(w)| = 0.707
926  Tp re = getReal(v * Td::ups());
927  Tp p1 = -re - Tp(2); // -re flips cutoff
928  mB1 = (p1 + sqrt(p1*p1 - Tp(1)));
929  mA0 = Tp(1) + mB1;
930  //printf("%g\n", mB1);
931  }
932  break;
933  case SMOOTHING:
934  lag(Tp(1)/mFreq);
935  break;
936  }
937
