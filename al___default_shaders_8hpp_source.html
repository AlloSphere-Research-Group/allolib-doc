52 inline std::string al_default_vert_shader_stereo_functions(bool is_omni) {
53  if (!is_omni) return R"(
54 vec4 stereo_displace(vec4 v, float e, float f) {
55  // eye to vertex distance
56  float l = sqrt((v.x - e) * (v.x - e) + v.y * v.y + v.z * v.z);
57  // absolute z-direction distance
58  float z = abs(v.z);
59  // x coord of projection of vertex on focal plane when looked from eye
60  float t = f * (v.x - e) / z;
61  // x coord of displaced vertex to make displaced vertex be projected on focal plane
62  // when looked from origin at the same point original vertex would be projected
63  // when looked form eye
64  v.x = z * (e + t) / f;
65  // set distance fromr origin to displaced vertex same as eye to original vertex
66  v.xyz = normalize(v.xyz);
67  v.xyz *= l;
68  return v;
69 })";
70  else return R"(
71 vec4 stereo_displace(vec4 v, float e, float r) {
72  vec3 OE = vec3(-v.z, 0.0, v.x); // eye direction, orthogonal to vertex vector
73  OE = normalize(OE); // but preserving +y up-vector
74  OE *= e; // set mag to eye separation
75  vec3 EV = v.xyz - OE; // eye to vertex
76  float ev = length(EV); // save length
77  EV /= ev; // normalize
78 
79  // coefs for polynomial t^2 + 2bt + c = 0
80  // derived from cosine law r^2 = t^2 + e^2 + 2tecos(theta)
81  // where theta is angle between OE and EV
82  // t is distance to sphere surface from eye
83  float b = -dot(OE, EV); // multiply -1 to dot product because
84  // OE needs to be flipped in direction
85  float c = e * e - r * r;
86  float t = -b + sqrt(b * b - c); // quadratic formula
87 
88  v.xyz = OE + t * EV; // direction from origin to sphere surface
89  v.xyz = ev * normalize(v.xyz); // normalize and set mag to eye-to-v distance
90  return v;
91 })";
92 }

249 #version 330
250 uniform vec4 col0;
