695 template <class Tv,class Tp>
696 Curve<Tv,Tp>& Curve<Tv,Tp>::set(Tp len, Tp crv, Tv start, Tv end){
697  static const Tp EPS = eps<Tp>();
698 
699  if(len == Tp(0)){ // if length is 0, return end value immediately
700  mEnd = end;
701  mMul = maxReal<Tp>();
702  mA = end;
703  mB = Tv(0);
704  return *this;
705  }
706 
707  // Avoid discontinuity when curve = 0 (a line)
708  if(crv < EPS && crv > -EPS){
709  crv = crv < Tp(0) ? -EPS : EPS;
710  }
711 
712  Tp crvOverLen = crv / len;
713 
714  if(crvOverLen < EPS && crvOverLen > -EPS){
715  crvOverLen = crvOverLen < Tp(0) ? -EPS : EPS;
716  crv = crvOverLen * len;
717  }
718 
719  /*
720  This algorithm uses an exponential curve in [0,1] to linearly interpolate
721  between the start and end values.
722 
723  1 - e^(cx)
724  y = ---------- * (end-start) + start
725  1 - e^c
726 
727  delta delta
728  = ------- + start - ------- e^(cx)
729  1 - e^c 1 - e^c
730  */
731 
732  mEnd = end;
733  mMul = exp(crvOverLen);
734  mA = (end-start) / (Tp(1) - exp(crv));
735  mB = mA / mMul;
736  mA+= start;
737  return *this;
738 }

801 template <int N,class Tv,class Tp,class Td>
802 inline Tv Env<N,Tv,Tp,Td>::operator()(){
803 
804  // Sustain stage:
805  if(sustained()){
806  return mLevels[mStage];
807  }
808 
809  // Interpolating along segment:
810  else if(mPos < mLen){
811  ++mPos;
812  return mCurve();
813  }
814 
815  // Just went past end of current segment and there are more left:
816  else if(mStage < size()){
817  ++mStage;
818  if(mLoop && done()) mStage=0;
819  if(!done()){
820  mPos = 0;
821  setLen(mStage);
822  int nextStage = mStage+1;
823  // If looping, ensure we wrap back around to first level
824  if(mLoop && (nextStage==size())) nextStage = 0;
825  mCurve.set(mLen, mCurves[mStage], mLevels[mStage], mLevels[nextStage]);
826 
827  // Immediately return start level of new stage
828  return (*this)();
829  }
830  }
831 
832  // Envelope is done:
833  return mLevels[mStage];
834 }
