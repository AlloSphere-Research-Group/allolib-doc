694 template <class Tv,class Tp>
695 Curve<Tv,Tp>& Curve<Tv,Tp>::set(Tp len, Tp crv, Tv start, Tv end){
696  static const Tp EPS = eps<Tp>();
697 
698  if(len == Tp(0)){ // if length is 0, return end value immediately
699  mEnd = end;
700  mMul = maxReal<Tp>();
701  mA = end;
702  mB = Tv(0);
703  return *this;
704  }
705 
706  // Avoid discontinuity when curve = 0 (a line)
707  if(crv < EPS && crv > -EPS){
708  crv = crv < Tp(0) ? -EPS : EPS;
709  }
710 
711  Tp crvOverLen = crv / len;
712 
713  if(crvOverLen < EPS && crvOverLen > -EPS){
714  crvOverLen = crvOverLen < Tp(0) ? -EPS : EPS;
715  crv = crvOverLen * len;
716  }
717 
718  /*
719  This algorithm uses an exponential curve in [0,1] to linearly interpolate
720  between the start and end values.
721 
722  1 - e^(cx)
723  y = ---------- * (end-start) + start
724  1 - e^c
725 
726  delta delta
727  = ------- + start - ------- e^(cx)
728  1 - e^c 1 - e^c
729  */
730 
731  mEnd = end;
732  mMul = exp(crvOverLen);
733  mA = (end-start) / (Tp(1) - exp(crv));
734  mB = mA / mMul;
735  mA+= start;
736  return *this;
737 }

800 template <int N,class Tv,class Tp,class Td>
801 inline Tv Env<N,Tv,Tp,Td>::operator()(){
802 
803  // Sustain stage:
804  if(sustained()){
805  return mLevels[mStage];
806  }
807 
808  // Interpolating along segment:
809  else if(mPos < mLen){
810  ++mPos;
811  return mCurve();
812  }
813 
814  // Just went past end of current segment and there are more left:
815  else if(mStage < size()){
816  ++mStage;
817  if(mLoop && done()) mStage=0;
818  if(!done()){
819  mPos = 0;
820  setLen(mStage);
821  int nextStage = mStage+1;
822  // If looping, ensure we wrap back around to first level
823  if(mLoop && (nextStage==size())) nextStage = 0;
824  mCurve.set(mLen, mCurves[mStage], mLevels[mStage], mLevels[nextStage]);
825 
826  // Immediately return start level of new stage
827  return (*this)();
828  }
829  }
830 
831  // Envelope is done:
832  return mLevels[mStage];
833 }
