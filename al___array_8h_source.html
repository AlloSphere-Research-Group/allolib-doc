299 static inline size_t allo_type_size(AlloTy ty) {
300  switch(ty) {
301  case AlloUInt8Ty: return sizeof(uint8_t);
302  case AlloUInt16Ty: return sizeof(uint16_t);
303  case AlloUInt32Ty: return sizeof(uint32_t);
304  case AlloUInt64Ty: return sizeof(uint64_t);
305  case AlloSInt8Ty: return sizeof(int8_t);
306  case AlloSInt16Ty: return sizeof(int16_t);
307  case AlloSInt32Ty: return sizeof(int32_t);
308  case AlloSInt64Ty: return sizeof(int64_t);
309  case AlloFloat32Ty: return sizeof(float);
310  case AlloFloat64Ty: return sizeof(double);
311  case AlloArrayTy: return sizeof(AlloArray);
312  case AlloPointer32Ty: return sizeof(int32_t);
313  case AlloPointer64Ty: return sizeof(int64_t);
314  default: return 0;
315  }
316 }

318 static inline double allo_type_tonumber(AlloTy ty, const char * ptr) {
319  switch(ty) {
320  case AlloUInt8Ty: return (double)(((uint8_t *)ptr)[0])/255.; /* UCHAR_MAX */
321  case AlloUInt16Ty: return (double)(((uint8_t *)ptr)[0])/65535.;
322  case AlloUInt32Ty: return (double)(((uint8_t *)ptr)[0])/(double)0xffffffff;
323  case AlloUInt64Ty: return (double)(((uint8_t *)ptr)[0])/(double)ULONG_MAX;
324  case AlloSInt8Ty: return 0.5+(double)(((uint8_t *)ptr)[0])/255.; /* UCHAR_MAX */
325  case AlloSInt16Ty: return 0.5+(double)(((uint8_t *)ptr)[0])/65535.;
326  case AlloSInt32Ty: return 0.5+(double)(((uint8_t *)ptr)[0])/(double)0xffffffff;
327  case AlloSInt64Ty: return 0.5+(double)(((uint8_t *)ptr)[0])/(double)ULONG_MAX;
328  case AlloFloat32Ty: return (double)(((float *)ptr)[0]);
329  case AlloFloat64Ty: return ((double *)ptr)[0];
330  default: return 0;
331  }
332 }

334 static inline void allo_type_fromnumber(AlloTy ty, double number, char * dst) {
335  switch(ty) {
336  case AlloUInt8Ty: *((uint8_t *)dst) = (uint8_t)(number * 255.); break;
337  case AlloUInt16Ty: *((uint16_t *)dst) = (uint16_t)(number * 65535.); break;
338  case AlloUInt32Ty: *((uint32_t *)dst) = (uint32_t)(number * (double)(0xffffffff)); break;
339  case AlloUInt64Ty: *((uint64_t *)dst) = (uint64_t)(number * (double)(ULONG_MAX)); break;
340 
341  case AlloSInt8Ty: *((int8_t *)dst) = (int8_t)((number-0.5) * 255.); break;
342  case AlloSInt16Ty: *((int16_t *)dst) = (int16_t)((number-0.5) * 65535.); break;
343  case AlloSInt32Ty: *((int32_t *)dst) = (int32_t)((number-0.5) * (double)(0xffffffff)); break;
344  case AlloSInt64Ty: *((int64_t *)dst) = (int64_t)((number-0.5) * (double)(ULONG_MAX)); break;
345  case AlloFloat32Ty: *((float *)dst) = (float)number; break;
346  case AlloFloat64Ty: *((double *)dst) = number; break;
347  default: *dst = 0;
348  }
349 }
