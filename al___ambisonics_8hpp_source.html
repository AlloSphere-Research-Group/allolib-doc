57  TODO: near-field compensation
58  acknowledges that the speakers are at a finite distance from the listener. This a simple Minimum Phase HP filter (on directional signals, not the W component) to deal with "proximity effect" because the speakers are 'near' the listener.
59 
60  f = c / (2 * PI * d) = 54.6 / d Hz c : speed of sound 343 m/s
61  d : speaker distance m
62 
63  Should be IIR, because it mostly affects LF.

426 inline void AmbiEncode::encode(float * ambiChans, int numFrames, int timeIndex, float timeSample) const {
427 
428  // "Iterate" through spherical harmonics using Duff's device.
429  // This requires only a simple jump per time sample.
430  #define CS(chanindex) case chanindex: ambiChans[chanindex*numFrames+timeIndex] += weights()[chanindex] * timeSample;
431  int ch = channels()-1;
432  switch(ch){
433  CS(15) CS(14) CS(13) CS(12) CS(11) CS(10) CS( 9) CS( 8)
434  CS( 7) CS( 6) CS( 5) CS( 4) CS( 3) CS( 2) CS( 1) CS( 0)
435  default:;
436  }
437  #undef CS
438 }

440 inline void AmbiEncode::encode(float * ambiChans, const float * input, int numFrames)
441 {
442  float * pAmbi = ambiChans; // non-interleaved ambi buffers, we can use fast pointer arithmetic
443 
444  for(int c=0; c<channels(); ++c){
445  const float * pInput = input;
446  float weight = weights()[c];
447  for(int i=0; i<numFrames; ++i){
448  *pAmbi++ += weight * *pInput++;
449  }
450  }
451 }
