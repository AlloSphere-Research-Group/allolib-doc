514

517  }

541  refCount()[mElems] = 1;

615 template<class T, class A>
616 void Ring<T,A>::readFrom(T * dst, uint32_t len, int32_t delay, uint32_t from) const{
617 
618  if(delay < 0) delay += int32_t(size());
619 
620  uint32_t begin = (uint32_t)scl::wrap(int32_t(from) - delay, int32_t(size()));
621 
622  // this ensures that we don't copy across the ring tap boundary
623  // we add one to maxLen because of a fence post anomaly
624  uint32_t maxLen = (begin < from ? (from - begin) : (from + (size() - begin))) + 1;
625  len = scl::min(len, maxLen);
626 
627  mem::copyFromRing(elems(), size(), begin, dst, len);
628 }
