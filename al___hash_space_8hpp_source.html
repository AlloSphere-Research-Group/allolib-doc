43  File description:
44  HashSpace is a way to detect object collisions using a voxel grid
45  The grid has a given resolution (no. voxel cells per side)
46 
47  It is optimized for densely packed points and querying for nearest neighbors
48  within given radii (results will be roughly sorted by distance).
49 
50  TODO: non-toroidal options
51  TODO: have query() automatically (insertion) sort results by distance
52  (perhaps use std::set instead of vector?)

363 // the maximum permissible value of radius is mDimHalf
364 // if int(inner^2) == int(outer^2), only 1 shell will be queried.
365 // TODO: non-toroidal version.
366 inline int HashSpace::Query :: operator()(const HashSpace& space, Vec3d center, double maxRadius, double minRadius) {
367  unsigned nres = 0;
368  double minr2 = minRadius*minRadius;
369  double maxr2 = maxRadius*maxRadius;
370  uint32_t iminr2 = std::max(uint32_t(0), uint32_t(minRadius*minRadius));
371  uint32_t imaxr2 = std::min(space.mMaxHalfD2, uint32_t(1 + (maxRadius+1)*(maxRadius+1)));
372  if (iminr2 < imaxr2) {
373  uint32_t cellstart = space.mDistanceToVoxelIndices[iminr2];
374  uint32_t cellend = space.mDistanceToVoxelIndices[imaxr2];
375  for (uint32_t i = cellstart; i < cellend; i++) {
376  uint32_t index = space.hash(center, space.mVoxelIndices[i]);
377  const Voxel& voxel = space.mVoxels[index];
378  // now add any objects in this voxel to the result...
379  Object * head = voxel.mObjects;
380  if (head) {
381  Object * o = head;
382  do {
383  // final check - float version:
384  Vec3d rel = space.wrapRelative(o->pos - center);
385  double d2 = rel.magSqr();
386  if (d2 >= minr2 && d2 <= maxr2) {
387  mObjects[nres].object = o;
388  mObjects[nres].distanceSquared = d2;
389  nres++;
390  }
391  o = o->next;
392  } while (o != head && nres < mMaxResults);
393  }
394  if(nres == mMaxResults) {
395  break;
396  }
397  }
398  }
399  //std::sort(mObjects.begin(), mObjects.end(), Result::compare);
400  return nres;
401 }

500 // wrapping floating point numbers is surprisingly complex
501 // because of rounding errors, behavior when near zero
502 // and other oddities
503 inline double HashSpace :: wrap(double x, double mod) {
504  if (mod) {
505  if (x > mod) {
506  // shift down
507  if (x > (mod*2.)) {
508  // multiple wraps:
509  double div = x / mod;
510  // get fract:
511  double divl = (long)div;
512  double fract = div - (double)divl;
513  return fract * mod;
514  } else {
515  // single wrap:
516  return x - mod;
517  }
518  } else if (x < 0.) {
519  // negative x, shift up
520  if (x < -mod) {
521  // multiple wraps:
522  double div = x / mod;
523  // get fract:
524  double divl = (long)div;
525  double fract = div - (double)divl;
526  double x1 = fract * mod;
527  return (x1 < 0.) ? x1 + mod : x1;
528  } else {
529  // single wrap:
530  return x + mod;
531  }
532  } else {
533  return x;
534  }
535  } else {
536  return 0.; // avoid divide by zero
537  }
538 }
