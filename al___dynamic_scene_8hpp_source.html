201 template <class F>
202 void ThreadPool::enqueue(F &&f, UpdateThreadFuncData &data) {
203  std::unique_lock<std::mutex> lock(queue_mutex);
204  tasks.emplace_back(std::pair<std::function<void(UpdateThreadFuncData)>,
205  UpdateThreadFuncData>(std::forward<F>(f), data));
206  cv_task.notify_one();
