119 inline char intensityToASCII(float v){
120  static const char map[] =
121  " .,;-~_+<>i!lI?/|)(1}{][rcvunxzjftLCJUYXZO0Qoahkbdpqwm*WMB8&%$#@";
122  //"$@B%8&WM#*oahkbdpqwmZO0QLCJUYXzcvunxrjft/\|()1{}[]?-_+~<>i!lI;:,\"^`'. ";
123 // 123456789.123456789.123456789.123456789.123456789.123456789.1234
124  static const int N = sizeof(map)-1;
125  v = v<0 ? 0 : (v>0.9999999f ? 0.9999999f : v);
126  return map[int(N*v)];
127 }
