80  Mesh &getTics() {
81  tics.reset();
82  Vec3f halfDim = dim / 2;
83  float ticLen = 0.1; // tic length multiplier
84  // x tics
85  for (int i = 0; i < dim.x * ticsPerUnit; i++) {
86  for (int z = 0; z < 2; z++) {
87  for (int y = -1; y < 2; y++) {
88  if (y != 0) {
89  float len = ticLen;
90  if (i % 5 == 0) len *= 2;
91  if (i % 10 == 0) len *= 1.5;
92  tics.vertex((i / ticsPerUnit) - halfDim.x, y * halfDim.y,
93  z * dim.z);
94  tics.vertex((i / ticsPerUnit) - halfDim.x,
95  y * halfDim.y - (len * y), z * dim.z);
96 
97  tics.vertex((i / ticsPerUnit) - halfDim.x, y * halfDim.y,
98  z * dim.z);
99  tics.vertex((i / ticsPerUnit) - halfDim.x, y * halfDim.y,
100  z * dim.z - (((z * 2) - 1) * len));
101  }
102  }
103  }
104  }
105  // y tics
106  for (int i = 0; i < dim.y * ticsPerUnit; i++) {
107  for (int z = 0; z < 2; z++) {
108  for (int x = -1; x < 2; x++) {
109  if (x != 0) {
110  float len = ticLen;
111  if (i % 5 == 0) len *= 2;
112  if (i % 10 == 0) len *= 1.5;
113  tics.vertex(x * halfDim.x, (i / ticsPerUnit) - halfDim.y,
114  z * dim.z);
115  tics.vertex(x * halfDim.x - (len * x),
116  (i / ticsPerUnit) - halfDim.y, z * dim.z);
117 
118  tics.vertex(x * halfDim.x, (i / ticsPerUnit) - halfDim.y,
119  z * dim.z);
120  tics.vertex(x * halfDim.x, (i / ticsPerUnit) - halfDim.y,
121  z * dim.z - (((z * 2) - 1) * len));
122  }
123  }
124  }
125  }
126  // z tics
127  for (int i = 0; i < dim.z * ticsPerUnit; i++) {
128  for (int z = 0; z < 2; z++) {
129  for (int x = -1; x < 2; x++) {
130  if (x != 0) {
131  float len = ticLen;
132  if (i % 5 == 0) len *= 2;
133  if (i % 10 == 0) len *= 1.5;
134  tics.vertex(z * dim.x - halfDim.x, x * halfDim.y,
135  (i / ticsPerUnit));
136  tics.vertex(z * dim.x - halfDim.x, x * halfDim.y - (len * x),
137  (i / ticsPerUnit));
138 
139  tics.vertex(z * dim.x - halfDim.x, x * halfDim.y,
140  (i / ticsPerUnit));
141  tics.vertex((z * dim.x - halfDim.x) - (((z * 2) - 1) * len),
142  x * halfDim.y, (i / ticsPerUnit));
143  }
144  }
145  }
146  }
147  tics.color(.4, .4, .4);
148  tics.primitive(Mesh::LINES);
149  tics.translate(cen.x, cen.y, cen.z - (dim.z / 2));
150  return tics;
151  }
