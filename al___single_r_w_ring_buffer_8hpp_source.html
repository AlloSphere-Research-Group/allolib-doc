136 inline size_t SingleRWRingBuffer :: write(const char * src, size_t sz) {
137  size_t space = writeSpace();
138  sz = sz > space ? space : sz;
139  if (sz == 0) return 0;
140 
141  size_t w = mWrite;
142  size_t end = w + sz;

144  if (end < mSize) {
