263 inline void RNGTaus::iterate(){
264  s1 = ((s1 & 0xfffffffe) << 18) ^ (((s1 << 6) ^ s1) >> 13);
265  s2 = ((s2 & 0xfffffff8) << 2) ^ (((s2 << 2) ^ s2) >> 27);

304 float gaussian(RNG& rng = rnd::gen){
305  float x1, x2, w, y1, y2;
306 
307  do{
308  x1 = uniS_float(rng);
309  x2 = uniS_float(rng);
310  w = x1 * x1 + x2 * x2;
311  } while( w >= 1.f );
