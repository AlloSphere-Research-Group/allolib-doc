164 template <int L_MAX=16>
165 class SphericalHarmonic{
166 public:

178  template <class T>
179  Complex<T> operator()(int l, int m, const Complex<T>& ctheta, const Complex<T>& cphi) const {
180  return coef(l,m) * al::legendreP(l, std::abs(m), cphi.r, cphi.i) * expim(m, ctheta);
181  }

183  template <class T>
184  static Complex<T> expim(int m, const Complex<T>& ctheta){
185  Complex<T> res = al::powN(ctheta, std::abs(m));
186  if(m < 0) res.i = -res.i;
187  return res;
188  }
189 
191  static double coef(int l, int m){ return l<=L_MAX ? coefTab(l,m) : coefCalc(l,m); }
192 
194  static const double& coefTab(int l, int m){ return LUT(l,m); }
195 
197  static double coefCalc(int l, int m){
198  int M = std::abs(m);
199  double res = ::sqrt((2*l + 1) / M_4PI) * al::factorialSqrt(l-M) / al::factorialSqrt(l+M);
200  return (m<0 && al::odd(M)) ? -res : res;
201  }
202 
203 private:
204  // this holds precomputed coefficients for each basis
205  static double& LUT(int l, int m){
206  static double t[L_MAX+1][L_MAX*2+1];
207  return t[l][m+L_MAX];
208  }
209 
210  static void createLUT(){
211  static bool make=true;
212  if(make){
213  make=false;
214  for(int l=0; l<=L_MAX; ++l){
215  for(int m=-L_MAX; m<=L_MAX; ++m){
216  double c=0;
217  // m must be in [-l,l]
218  if(std::abs(m) <= l) c = coefCalc(l,m);
219  LUT(l, m) = c;
220  }
221  }
222  }
223  }
224 };
