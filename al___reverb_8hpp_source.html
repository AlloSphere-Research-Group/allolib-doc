60 template <int N, class T>
61 class StaticDelayLine {
62  public:
63  StaticDelayLine() : mPos(0) { zero(); }
64 
66  static int size() { return N; }
67 
69  const T& back() const { return mBuf[indexBack()]; }
70 
72  int indexBack() const {
73  int i = pos() + 1;
74  return (i < size()) ? i : 0;
75  }
76 
78  int pos() const { return mPos; }
79 
81  const T& read(int i) const {
82  int ind = pos() - i;
83  if (ind < 0) ind += size();
84  // else if(ind >= size()) ind -= size();
85  return mBuf[ind];
86  }
87 
89  void write(const T& v) {
90  mBuf[pos()] = v;
91  ++mPos;
92  if (mPos >= size()) mPos = 0;
93  }
94 
96  T operator()(const T& v) {
97  T r = mBuf[pos()];
98  write(v);
99  return r;
100  }
101 
104  T comb(const T& v, const T& ffd, const T& fbk) {
105  T d = mBuf[pos()];
106  T r = v + d * fbk;
107  write(r);
108  return d + r * ffd;
109  }
110 
113  T allpass(const T& v, const T& ffd) { return comb(v, ffd, -ffd); }
114 
116  void zero() { ::memset(&mBuf, 0, sizeof(mBuf)); }
117 
118  protected:
119  int mPos;
120  T mBuf[N];
121 };
