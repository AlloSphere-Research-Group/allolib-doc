183  Rayd transformRayLocal(const Rayd &ray) {
184  Matrix4d t, r, s;
185  Matrix4d model = t.translation(pose.get().pos()) *
186  r.fromQuat(pose.get().quat()) * s.scaling(scaleVec.get());
187  Matrix4d invModel = Matrix4d::inverse(model);

250  bool onEvent(PickEvent e, Hit h) {
251  switch (e.type) {
252  case Point:
253  if (hover.get() != h.hit)
254  hover = h.hit; // setting value propagates via OSC, so only set if
255  // there is a change
256  return h.hit;
257  break;
258 
259  case Pick:
260  if (h.hit) {
261  prevPose.set(pose.get());
262  selectDist = h.t;
263  selectPos = h();
264  selectOffset = pose.get().pos() - h(); // * scaleVec.get();
265  }
266  if (selected.get() != h.hit)
267  selected = h.hit; // to avoid triggering change callback if no change
268 
269  return h.hit;
270  break;
271 
272  case PickPose:
273  prevPose.set(pose.get());
274  selectQuat.set(e.pose.quat());
275  selectOffset.set(pose.get().pos() - e.pose.pos());
276 
277  if (selected.get() != true) selected = true;
278 
279  return true;
280  break;
281 
282  case Drag:
283  case TranslateRay:
284  if (selected.get()) {
285  Vec3f newPos = h.ray(selectDist) + selectOffset;
286  if (parent && (parent->containChildren || containedChild)) {
287  auto *p = dynamic_cast<PickableBB *>(parent);
288  if (p) {
289  // std::cout << "pre: " << newPos << std::endl;
290 
291  newPos = min(newPos, p->bb.max - scaleVec.get() * bb.dim / 2);
292  // std::cout << "mid: " << newPos << std::endl;
293 
294  newPos = max(newPos, p->bb.min + scaleVec.get() * bb.dim / 2);
295 
296  // std::cout << "post: " << newPos << std::endl;
297  // std::cout << "min: " << p->bb.min << " max: " << p->bb.max << "
298  // scale: " << scaleVec.get() << " hdim: " << bb.dim / 2 <<
299  // std::endl;
300  // newPos -= bb.dim / 2;
301  }
302  }
303  pose = Pose(newPos, pose.get().quat());
304  return true;
305  } else
306  return false;
307  break;
308 
309  case RotateRay:
310  if (selected.get()) {
311  Vec3f dir = h.ray(selectDist) - selectPos;
312  Quatf q = Quatf().fromEuler(dir.x * 0.5f, -dir.y * 0.5f, 0);
313 
314  Vec3f p1 = transformVecWorld(bb.cen);
315  pose.setQuat(q * prevPose.quat());
316  Vec3f p2 = transformVecWorld(bb.cen);
317  pose.setPos(pose.get().pos() + p1 - p2);
318  return true;
319  } else
320  return false;
321  break;
322 
323  case RotatePose:
324  if (selected.get()) {
325  Quatf diff =
326  e.pose.quat() *
327  selectQuat
328  .inverse(); // diff * q0 = q1 --> diff = q1 * q0.inverse
329 
330  Vec3f p1 = transformVecWorld(bb.cen);
331  pose.setQuat(diff * prevPose.quat());
332  Vec3f p2 = transformVecWorld(bb.cen);
333  // pose.setPos(e.pose.pos()+selectOffset + p1-p2);
334  pose.setPos(pose.get().pos() + p1 - p2);
335  return true;
336  } else
337  return false;
338  break;
339 
340  case Scale:
341  if (selected.get()) {
342  Vec3f p1 = transformVecWorld(bb.cen);
343  scale = scale + e.amount * 0.01 * scale;
344  if (scale < 0.0005) scale = 0.0005;
345  // scaleVec.set(scale)
346  Vec3f p2 = transformVecWorld(bb.cen);
347  pose.setPos(pose.get().pos() + p1 - p2);
348 
349  return true;
350  } else
351  return false;
352  break;
353 
354  case Unpick:
355  if (!hover.get() && selected.get()) selected = false;
356  return false;
357  break;
358 
359  default:
360  break;
361  }
362  }

428  // thanks to
429  // http://zeuxcg.org/2010/10/17/aabb-from-obb-with-component-wise-abs/

431  Matrix4d model = t.translation(pose.get().pos()) *
