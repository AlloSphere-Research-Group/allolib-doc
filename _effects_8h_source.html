143 template <unsigned N, class T=gam::real>
144 class ChebyN{
145 public:
146  T c[N];
147 
148  ChebyN(const T& fundAmp = T(1)){ zero(); c[0]=fundAmp; }
149 
151  T operator()(T i0) const { return i0*c[0] + wet(i0); }
152 
154  T wet(T i0) const {
155 // T d1 = i0 * T(2); // Chebyshev polynomial of 2nd kind
156  T d1 = i0; // Chebyshev polynomial of 1st kind
157  T d2 = T(1);
158  T b1 = T(2) * i0;
159 
160  T o0 = T(0);
161  for(unsigned i=1; i<N; ++i){
162  T d0 = b1 * d1 - d2;
163  d2 = d1;
164  d1 = d0;
165  o0 += d0 * c[i];
166  }
167  return o0;
168  }
169 
171  unsigned size() const { return N; }
172 
174  T& coef(int i){ return c[i]; }
175 
177  template <class V>
178  ChebyN& set(const V* weights){
179  for(unsigned i=0; i<N; ++i) c[i] = weights[i];
180  return *this;
181  }
182 
184  ChebyN& zero(){
185  for(unsigned i=0; i<N; ++i) c[i] = T(0);
186  return *this;
187  }
188 };
