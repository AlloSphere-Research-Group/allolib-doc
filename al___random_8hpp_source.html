318 inline void Tausworthe::seed(uint32_t v1, uint32_t v2, uint32_t v3, uint32_t v4){
319  //printf("%u %u %u %u\n", v1, v2, v3, v4);
320  v1 & 0xffffffe ? s1 = v1 : s1 = ~v1;
321  v2 & 0xffffff8 ? s2 = v2 : s2 = ~v2;
322  v3 & 0xffffff0 ? s3 = v3 : s3 = ~v3;
323  v4 & 0xfffff80 ? s4 = v4 : s4 = ~v4;
324 }

326 inline void Tausworthe::iterate(){
327  s1 = ((s1 & 0xfffffffe) << 18) ^ (((s1 << 6) ^ s1) >> 13);
328  s2 = ((s2 & 0xfffffff8) << 2) ^ (((s2 << 2) ^ s2) >> 27);
329  s3 = ((s3 & 0xfffffff0) << 7) ^ (((s3 << 13) ^ s3) >> 21);
330  s4 = ((s4 & 0xffffff80) << 13) ^ (((s4 << 3) ^ s4) >> 12);
331 }

337  T w;
338  do{

348 // Box-Muller transform
349 // Box, G. and Muller, M. A note on the generation of normal deviates.
350 // Ann. Math. Slat. 28, (1958).
351 //
352 // http://en.wikipedia.org/wiki/Box–Muller_transform
353 template <class RNG>
354 template <class T> void Random<RNG>::normal(T& y1, T& y2){
355  float x1, x2, w;
356 
357  // Search for point within unit circle using sample-reject.
358  // This will pass with probability π/4 = ~0.785.
359  do{
360  x1 = uniformS();
361  x2 = uniformS();
362  w = x1 * x1 + x2 * x2;
363  } while(w >= 1.f);
364 
365  // perform inverse Gaussian function mapping
366  w = std::sqrt((-2.f * std::log(w)) / w);
367  y1 = T(x1 * w);
368  y2 = T(x2 * w);
369 }
