218  Mat4f get_rotation_matrix(Vec3f axis, float angle) {
219  Mat4f r;
220  float cosT = cos(angle);
221  float sinT = sin(angle);
222  float X = axis.x, Y = axis.y, Z = axis.z;
223  r.set(cosT + X * X * (1 - cosT), X * Y * (1 - cosT) - Z * sinT,
224  X * Z * (1 - cosT) + Y * sinT, 0, Y * X * (1 - cosT) + Z * sinT,
225  cosT + Y * Y * (1 - cosT), Y * Z * (1 - cosT) - X * sinT, 0,
226  Z * X * (1 - cosT) - Y * sinT, Z * Y * (1 - cosT) + X * sinT,
227  cosT + Z * Z * (1 - cosT), 0, 0, 0, 0, 1);
228  return r;
229  }
