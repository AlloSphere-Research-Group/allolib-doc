316  // temporary hack because the one in al_Function gave a bad result
317  // for e.g. wrap<double>(-64.0, -32.0);
318  template<typename T>
319  static T wrap(T v, const T hi=T(1.), const T lo=T(0.)){
320  if(lo == hi) return lo;
321  //if(v >= hi){
322  if(!(v < hi)){
323  T diff = hi - lo;
324  v -= diff;
325  if(!(v < hi)) v -= diff * (T)(uint32_t)((v - lo)/diff);
326  }
327  else if(v < lo){
328  T diff = hi - lo;
329  v += diff;
330  if(v < lo) v += diff * (T)(uint32_t)(((lo - v)/diff) + 1);
331  if(v==diff) return lo;
332  }
333  return v;
334  }

417 // linear interpolated lookup (virtual array index)
418 // reads the linearly interpolated plane values into val array
419 template<typename T> inline void Array::read_interp(T * val, double x) const {
420  x = wrap<double>(x, (double)header.dim[0], 0.);
421  // convert 0..1 field indices to 0..(d-1) cell indices
422  const unsigned xa = (const unsigned)DOUBLE_FLOOR(x);
423  unsigned xb = xa+1; if (xb == header.dim[0]) xb = 0;
424  // get the normalized 0..1 interp factors, of x,y,z:
425  double faaa = DOUBLE_FRAC(x);
426  double fbaa = 1.f - faaa;
427  // get the cell addresses for each neighbor:
428  T * paaa = cell<T>(xa);
429  T * pbaa = cell<T>(xb);
430  // for each plane of the field, do the interp:
431  for (uint8_t p=0; p<header.components; p++) {
432  val[p] = (paaa[p] * faaa) + (pbaa[p] * fbaa);
433  }
434 }

436 template<typename T> inline void Array::read_interp(T * val, double x, double y) const {
437  x = wrap<double>(x, (double)header.dim[0], 0.);
438  y = wrap<double>(y, (double)header.dim[1], 0.);
439  // convert 0..1 field indices to 0..(d-1) cell indices
440  const unsigned xa = (const unsigned)DOUBLE_FLOOR(x);
441  const unsigned ya = (const unsigned)DOUBLE_FLOOR(y);
442  unsigned xb = xa+1; if (xb == header.dim[0]) xb = 0;
443  unsigned yb = ya+1; if (yb == header.dim[1]) yb = 0;
444  // get the normalized 0..1 interp factors, of x,y,z:
445  double xbf = DOUBLE_FRAC(x);
446  double xaf = 1.f - xbf;
447  double ybf = DOUBLE_FRAC(y);
448  double yaf = 1.f - ybf;
449  // get the interpolation corner weights:
450  double faaa = xaf * yaf;
451  double faba = xaf * ybf;
452  double fbaa = xbf * yaf;
453  double fbba = xbf * ybf;
454  // get the cell addresses for each neighbor:
455  T * paaa = cell<T>(xa, ya);
456  T * paba = cell<T>(xa, yb);
457  T * pbaa = cell<T>(xb, ya);
458  T * pbba = cell<T>(xb, yb);
459  // for each plane of the field, do the interp:
460  for (uint8_t p=0; p<header.components; p++) {
461  val[p] = (paaa[p] * faaa) +
462  (pbaa[p] * fbaa) +
463  (paba[p] * faba) +
464  (pbba[p] * fbba);
465  }
466 }

468 template<typename T> inline void Array::read_interp(T * val, double x, double y, double z) const {
469  x = wrap<double>(x, (double)header.dim[0], 0.);
470  y = wrap<double>(y, (double)header.dim[1], 0.);
471  z = wrap<double>(z, (double)header.dim[2], 0.);
472  // convert 0..1 field indices to 0..(d-1) cell indices
473  const unsigned xa = (const unsigned)DOUBLE_FLOOR(x);
474  const unsigned ya = (const unsigned)DOUBLE_FLOOR(y);
475  const unsigned za = (const unsigned)DOUBLE_FLOOR(z);
476  unsigned xb = xa+1; if (xb == header.dim[0]) xb = 0;
477  unsigned yb = ya+1; if (yb == header.dim[1]) yb = 0;
478  unsigned zb = za+1; if (zb == header.dim[2]) zb = 0;
479  // get the normalized 0..1 interp factors, of x,y,z:
480  double xbf = DOUBLE_FRAC(x);
481  double xaf = 1.f - xbf;
482  double ybf = DOUBLE_FRAC(y);
483  double yaf = 1.f - ybf;
484  double zbf = DOUBLE_FRAC(z);
485  double zaf = 1.f - zbf;
486  // get the interpolation corner weights:
487  double faaa = xaf * yaf * zaf;
488  double faab = xaf * yaf * zbf;
489  double faba = xaf * ybf * zaf;
490  double fabb = xaf * ybf * zbf;
491  double fbaa = xbf * yaf * zaf;
492  double fbab = xbf * yaf * zbf;
493  double fbba = xbf * ybf * zaf;
494  double fbbb = xbf * ybf * zbf;
495  // get the cell addresses for each neighbor:
496  T * paaa = cell<T>(xa, ya, za);
497  T * paab = cell<T>(xa, ya, zb);
498  T * paba = cell<T>(xa, yb, za);
499  T * pabb = cell<T>(xa, yb, zb);
500  T * pbaa = cell<T>(xb, ya, za);
501  T * pbab = cell<T>(xb, ya, zb);
502  T * pbba = cell<T>(xb, yb, za);
503  T * pbbb = cell<T>(xb, yb, zb);
504  // for each plane of the field, do the 3D interp:
505  for (size_t p=0; p<header.components; p++) {
506  val[p] = (paaa[p] * faaa) +
507  (pbaa[p] * fbaa) +
508  (paba[p] * faba) +
509  (paab[p] * faab) +
510  (pbab[p] * fbab) +
511  (pabb[p] * fabb) +
512  (pbba[p] * fbba) +
513  (pbbb[p] * fbbb);
514  }
515 }

548 // linear interpolated write (virtual array index)
549 // writes the linearly interpolated plane values from val array into array
550 template<typename T> inline void Array::write_interp(const T* val, double x) {
551  x = wrap<double>(x, (double)header.dim[0], 0.);
552  const unsigned xa = (const unsigned)DOUBLE_FLOOR(x);
553  unsigned xb = xa+1; if (xb == header.dim[0]) xb = 0;
554  // get the normalized 0..1 interp factors, of x,y,z:
555  double xbf = DOUBLE_FRAC(x);
556  double xaf = 1.f - xbf;
557  // get the interpolation corner weights:
558  double faaa = xaf;
559  double fbaa = xbf;
560  T * paaa = cell<T>(xa);
561  T * pbaa = cell<T>(xb);
562  // for each plane of the field, do the 3D interp:
563  for (uint8_t p=0; p<header.components; p++) {
564  T tmp = val[p];
565  paaa[p] += tmp * faaa;
566  pbaa[p] += tmp * fbaa;

570  x = wrap<double>(x, (double)header.dim[0], 0.);
571  y = wrap<double>(y, (double)header.dim[1], 0.);
572  const unsigned xa = (const unsigned)DOUBLE_FLOOR(x);
573  const unsigned ya = (const unsigned)DOUBLE_FLOOR(y);
574  unsigned xb = xa+1; if (xb == header.dim[0]) xb = 0;
575  unsigned yb = ya+1; if (yb == header.dim[1]) yb = 0;
576  // get the normalized 0..1 interp factors, of x,y,z:
577  double xbf = DOUBLE_FRAC(x);
578  double xaf = 1.f - xbf;
579  double ybf = DOUBLE_FRAC(y);
580  double yaf = 1.f - ybf;
581  // get the interpolation corner weights:
582  double faaa = xaf * yaf;
583  double faba = xaf * ybf;
584  double fbaa = xbf * yaf;
585  double fbba = xbf * ybf;

587  T * paba = cell<T>(xa, yb);
588  T * pbaa = cell<T>(xb, ya);
589  T * pbba = cell<T>(xb, yb);
