163  int numChannels = channels();
164 
165  if(1 == numChannels) return readAll(dst);
166 
167  // Allocate memory for deinterleaving. Don't know of any in-place methods.
168  T * temp = new T[samples()];
169  int framesRead = 0;
170 
171  if(temp){
172  framesRead = readAll(temp);
173  if(framesRead == frames()){
174  if(2 == numChannels) mem::deinterleave2(dst, temp, frames());
175  else mem::deinterleave(dst, temp, frames(), numChannels);
176  }
177  delete[] temp;
178  }
179  return framesRead;
180 }
