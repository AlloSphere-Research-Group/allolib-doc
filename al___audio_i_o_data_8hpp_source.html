72 // Utility function to deinterleave samples
73 template <class T>
74 static void deinterleave(T* dst, const T* src, int numFrames, int numChannels) {
75  int numSamples = numFrames * numChannels;
76  for (int c = 0; c < numChannels; c++) {
77  for (int i = c; i < numSamples; i += numChannels) {
78  *dst++ = src[i];
79  }
80  }
81 }
82 
84 template <class T>
85 static void interleave(T* dst, const T* src, int numFrames, int numChannels) {
86  int numSamples = numFrames * numChannels;
87  for (int c = 0; c < numChannels; c++) {
88  for (int i = c; i < numSamples; i += numChannels) {
89  dst[i] = *src++;
90  }
91  }
92 }
