71 template <class T>
72 static void deinterleave(T* dst, const T* src, int numFrames, int numChannels) {
73  int numSamples = numFrames * numChannels;
74  for (int c = 0; c < numChannels; c++) {
75  for (int i = c; i < numSamples; i += numChannels) {
76  *dst++ = src[i];
77  }
78  }
79 }
