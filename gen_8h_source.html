318  void set(double length, T offset=0.001){
319  using namespace std;
320  length *= 0.5;
321  mul2= pow(offset, T(2.)/(length*length)); // b^2
322  mul1= pow(T(1.)/mul2, T(length+0.5)); // rewind from center of window
323  val = offset;
324  }
325 
327  T operator()(){
328  T res = val;
329  mul1 *= mul2;
330  val *= mul1;
331  return res;
332  }
333 
335  bool done(float thresh=0.001) const {
336  // slope negative and value < threshold
337  return (gam::magSqr(mul1) < 1.) && (gam::magSqr(val) < thresh*thresh);
338  }
339 
340  double length() const {
341  /*m1 = (1/m2)^(L+0.5)
342  log(m1) = (L+0.5)log(1/m2)
343  L = log(m1)/log(1/m2) - 0.5
344  L = log(m1)/(-log(m2)) - 0.5*/
345  return (log(gam::norm(mul1))/-log(gam::norm(mul2)) - 0.5)*2;
346  }
347 
348  T offset() const {
349  /*m2 = O^(2/LL)
350  m2^(LL/2) = O*/
351  double L = length();
352  return pow(mul2, T(L*L*0.125));
353  }
