309  void set(double length, T offset=0.001){
310  using namespace std;
311  length *= 0.5;
312  mul2= pow(offset, T(2.)/(length*length)); // b^2
313  mul1= pow(T(1.)/mul2, T(length+0.5)); // rewind from center of window
314  val = offset;
315  }
316 
318  T operator()(){
319  T res = val;
320  mul1 *= mul2;
321  val *= mul1;
322  return res;
323  }
324 
326  bool done(float thresh=0.001) const {
327  // slope negative and value < threshold
328  return (gam::magSqr(mul1) < 1.) && (gam::magSqr(val) < thresh*thresh);
329  }
330 
331  double length() const {
332  /*m1 = (1/m2)^(L+0.5)
333  log(m1) = (L+0.5)log(1/m2)
334  L = log(m1)/log(1/m2) - 0.5
335  L = log(m1)/(-log(m2)) - 0.5*/
336  return (log(gam::norm(mul1))/-log(gam::norm(mul2)) - 0.5)*2;
337  }
338 
339  T offset() const {
340  /*m2 = O^(2/LL)
341  m2^(LL/2) = O*/
342  double L = length();
343  return pow(mul2, T(L*L*0.125));
344  }
