267 template <class T>
268 void Isosurface::generate(const T * vals){
269  inBox(true);
270  begin();
271 
272  int Nx = mNF[0];
273  int Nxy = Nx*mNF[1];
274 
275  // iterate through cubes (not field points)
276  //for(int z=0; z < mNF[2]-1; ++z){
277  // support transparency (assumes higher indices are farther away)
278  for(int z=mNF[2]-2; z>=0; --z){
279  int z0 = z *Nxy;
280  int z1 =(z+1)*Nxy;
281  for(int y=0; y < mNF[1]-1; ++y){
282  int y0 = y *Nx;
283  int y1 =(y+1)*Nx;
284 
285  int z0y0 = z0+y0;
286  int z0y1 = z0+y1;
287  int z1y0 = z1+y0;
288  int z1y1 = z1+y1;
289 
290  int z0y0_1 = z0y0+1;
291  int z0y1_1 = z0y1+1;
292  int z1y0_1 = z1y0+1;
293  int z1y1_1 = z1y1+1;
294 
295  for(int x=0; x < mNF[0]-1; ++x){
296 
297  float v8[] = {
298  float(vals[z0y0 + x]), float(vals[z0y0_1 + x]),
299  float(vals[z0y1 + x]), float(vals[z0y1_1 + x]),
300  float(vals[z1y0 + x]), float(vals[z1y0_1 + x]),
301  float(vals[z1y1 + x]), float(vals[z1y1_1 + x])
302  };
303 
304  int i3[] = {x,y,z};
305 
306  addCell(i3, v8);
307  }
308  }
309  }
310 
311  end();
312 }
