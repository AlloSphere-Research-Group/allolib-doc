197  float getNextSample(Pose listeningPose) {
198  float s = 0.0f;
199  double dist =
200  listeningPose.vec().mag(); // Compute distance in world-space units
201  Vec3d relDirection = posHistory()[0] - listeningPose.vec();
202 
203  Quatd srcRot = listeningPose.quat();
204  relDirection = srcRot.rotate(relDirection);
205  double distanceToSample = 0;
206  if (dopplerType() == DOPPLER_SYMMETRICAL) {
207  distanceToSample = mSampleRate / mSpeedOfSound;
208  } else if (dopplerType() == DOPPLER_PHYSICAL) {
209  // FIXME AC Can we use the current pose here for distance or should we
210  // calculate from listener history?
211  double prevDistance = (posHistory()[1] - listeningPose.vec()).mag();
212  double sourceVel =
213  (dist - prevDistance) *
214  mSampleRate; // positive when moving away, negative moving toward
215  // if(sourceVel == -mSpeedOfSound) sourceVel -=
216  //0.001; //prevent divide by 0 / inf freq
217  double sumSpeed = mSpeedOfSound + sourceVel;
218  if (sumSpeed < 0.001) {
219  sumSpeed = 0.001;
220  }
221 
222  distanceToSample = fabs(mSampleRate / sumSpeed);
223  }
224  updateHistory();
225 
226  // Compute how many samples ago to read from buffer
227  // Start with time delay due to speed of sound
228  double samplesAgo = dist * distanceToSample;
229 
230  // // Add on time delay (in samples) - only needed if the source is
231  //rendered per buffer if(!usePerSampleProcessing())
232  samplesAgo += mFrameCounter++;
233 
234  // Is our delay line big enough?
235  if (samplesAgo <= maxIndex()) {
236  double gain = attenuation(dist);
237  // std::cout << dist << ":" << gain << std::endl;
238 
239  // This seemed to get the same sample per block
240  // float s = src.readSample(samplesAgo) * gain;
241 
242  // reading samplesAgo-i causes a discontinuity
243  s = readSample(samplesAgo) * gain;
244 
245  // s = src.presenceFilter(s); //TODO: causing stopband ripple here, why?
246  }
247  return s;
248  }

271  float readSample(double index) const {
272  int index0 = index;
273  float a = mSound.read(index0);
274  float b = mSound.read(index0 + 1);
275  float frac = index - index0;
276  // return ipl::linear(frac, a, b);
277 
278  float a0 = mSound.read(index0 - 1);
279  float b1 = mSound.read(index0 + 2);
280  return ipl::cubic(frac, a0, a, b, b1);
281  }
