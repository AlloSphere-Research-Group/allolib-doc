195  float getNextSample(Pose listeningPose) {
196 
197  float s = 0.0f;
198  double dist = listeningPose.vec().mag(); //Compute distance in world-space units
199  Vec3d relDirection = posHistory()[0] - listeningPose.vec();
200 
201  Quatd srcRot = listeningPose.quat();
202  relDirection = srcRot.rotate(relDirection);
203  double distanceToSample = 0;
204  if(dopplerType() == DOPPLER_SYMMETRICAL) {
205  distanceToSample = mSampleRate / mSpeedOfSound;
206  } else if(dopplerType() == DOPPLER_PHYSICAL) {
207  // FIXME AC Can we use the current pose here for distance or should we calculate from listener history?
208  double prevDistance = (posHistory()[1] - listeningPose.vec()).mag();
209  double sourceVel = (dist - prevDistance)*mSampleRate; //positive when moving away, negative moving toward
210 // if(sourceVel == -mSpeedOfSound) sourceVel -= 0.001; //prevent divide by 0 / inf freq
211  double sumSpeed = mSpeedOfSound + sourceVel;
212  if (sumSpeed < 0.001) { sumSpeed = 0.001; }
213 
214  distanceToSample = fabs(mSampleRate / sumSpeed);
215  }
216  updateHistory();
217 
218  // Compute how many samples ago to read from buffer
219  // Start with time delay due to speed of sound
220  double samplesAgo = dist * distanceToSample;
221 
222 // // Add on time delay (in samples) - only needed if the source is rendered per buffer
223 // if(!usePerSampleProcessing())
224  samplesAgo += mFrameCounter++;
225 
226  // Is our delay line big enough?
227  if(samplesAgo <= maxIndex()){
228  double gain = attenuation(dist);
229 // std::cout << dist << ":" << gain << std::endl;
230 
231  //This seemed to get the same sample per block
232  // float s = src.readSample(samplesAgo) * gain;
233 
234  //reading samplesAgo-i causes a discontinuity
235  s = readSample(samplesAgo) * gain;
236 
237  // s = src.presenceFilter(s); //TODO: causing stopband ripple here, why?
238 
239  }
240  return s;
241  }
