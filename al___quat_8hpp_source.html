371 inline Quat<T>& Quat<T> :: normalize() {
372  T unit = magSqr();
373  if(unit*unit < eps()){
374  // unit too close to epsilon, set to default transform
375  setIdentity();
376  }

407 template<typename T>
408 Quat<T> Quat<T> :: pow(T expo) const {
409  T m = mag();
410  T w_m = w / m;
411 
412  // Is q/|q| close to (±1,0,0,0)?
413  // If so, then quaternion cannot be "rotated", so just scale w component.
414  if(w_m > justUnder1<T>() || w_m < -justUnder1<T>()){
415  return Quat<T>((w>=T(0)?T(1):T(-1)) * ::pow(m, expo), 0, 0, 0);
416  }
417 
418  T theta = ::acos(w_m);
419  Vec<3,T> imag = Vec<3,T>(x,y,z) / (m * ::sin(theta));
420  imag *= ::sin(expo*theta);
421  return Quat(::cos(expo*theta), imag) * ::pow(m, expo);
422 }

441 template<typename T>
442 Quat<T>& Quat<T>::fromEuler(const T& az, const T& el, const T& ba){
443  /*http://vered.rose.utoronto.ca/people/david_dir/GEMS/GEMS.html
444  Converting from Euler angles to a quaternion is slightly more tricky, as the
445  order of operations must be correct. Since you can convert the Euler angles
446  to three independent quaternions by setting the arbitrary axis to the
447  coordinate axes, you can then multiply the three quaternions together to
448  obtain the final quaternion.
449 
450  So if you have three Euler angles (a, e, b), then you can form three
451  independent quaternions
452 
453  Qx = [ cos(e/2), (sin(e/2), 0, 0)]
454  Qy = [ cos(a/2), (0, sin(a/2), 0)]
455  Qz = [ cos(b/2), (0, 0, sin(b/2))]
456 
457  and the final quaternion, using YXZ ordering, is obtained by (Qy * Qx) * Qz.
458  */
459 
460  T c1 = cos(az * T(0.5));
461  T c2 = cos(el * T(0.5));
462  T c3 = cos(ba * T(0.5));
463  T s1 = sin(az * T(0.5));
464  T s2 = sin(el * T(0.5));
465  T s3 = sin(ba * T(0.5));
466 
467 /*
468  (w,x,y,z) (a,b,c,d)
469  w*a - x*b - y*c - z*d,
470  w*b + x*a + y*d - z*c,
471  w*c + y*a + z*b - x*d,
472  w*d + z*a + x*c - y*b
473 
474  (c1, 0,s1,0) (c2, s2,0,0)
475  c1*c2 - 0 - 0 - 0
476  c1*s2 + 0 + 0 + 0
477  0 + s1*c2 + 0 + 0
478  0 + 0 + 0 - s1*s2
479 
480  (tw, tx, ty, tz) (c3, 0,0,s3)
481  tw*c3 - 0 - 0 - tz*s3
482  0 + tx*c3 + ty*s3 + 0
483  0 + ty*c3 + 0 - tx*s3
484  tw*s3 + tz*c3 + 0 - 0
485 */
486  // equiv Q1 = Qy * Qx; // since many terms are zero
487  T tw = c1*c2;
488  T tx = c1*s2;
489  T ty = s1*c2;
490  T tz =-s1*s2;
491 
492  // equiv Q2 = Q1 * Qz; // since many terms are zero
493  w = tw*c3 - tz*s3;
494  x = tx*c3 + ty*s3;
495  y = ty*c3 - tx*s3;
496  z = tw*s3 + tz*c3;
497 
498  //return normalize();
499  return *this;
500 }

502 template<typename T>
503 Quat<T>& Quat<T> :: fromMatrix(const T * m) {
504  T trace = m[0]+m[5]+m[10];
505  w = sqrt(1. + trace)*0.5;
506 
507  if(trace > 0.) {
508  x = (m[9] - m[6])/(4.*w);
509  y = (m[2] - m[8])/(4.*w);
510  z = (m[4] - m[1])/(4.*w);
511  }
512  else {
513  if(m[0] > m[5] && m[0] > m[10]) {
514  // m[0] is greatest
515  x = sqrt(1. + m[0]-m[5]-m[10])*0.5;
516  w = (m[9] - m[6])/(4.*x);
517  y = (m[4] + m[1])/(4.*x);
518  z = (m[8] + m[2])/(4.*x);
519  }
520  else if(m[5] > m[0] && m[5] > m[10]) {
521  // m[1] is greatest
522  y = sqrt(1. + m[5]-m[0]-m[10])*0.5;
523  w = (m[2] - m[8])/(4.*y);
524  x = (m[4] + m[1])/(4.*y);
525  z = (m[9] + m[6])/(4.*y);
526  }
527  else { //if(m[10] > m[0] && m[10] > m[5]) {
528  // m[2] is greatest
529  z = sqrt(1. + m[10]-m[0]-m[5])*0.5;
530  w = (m[4] - m[1])/(4.*z);
531  x = (m[8] + m[2])/(4.*z);
532  y = (m[9] + m[6])/(4.*z);
533  }
534  }
535  return *this;
536 }

540  T trace = m[0]+m[5]+m[10];
541  w = sqrt(1. + trace)*0.5;
542 
543  if(trace > 0.) {
544  x = (m[6] - m[9])/(4.*w);

552  w = (m[6] - m[9])/(4.*x);
553  y = (m[1] + m[4])/(4.*x);
554  z = (m[2] + m[8])/(4.*x);
555  }
556  else if(m[5] > m[0] && m[5] > m[10]) {
557  // m[1] is greatest
558  y = sqrt(1. + m[5]-m[0]-m[10])*0.5;
559  w = (m[8] - m[2])/(4.*y);
560  x = (m[1] + m[4])/(4.*y);
561  z = (m[6] + m[9])/(4.*y);
562  }
563  else { //if(m[10] > m[0] && m[10] > m[5]) {
564  // m[2] is greatest
565  z = sqrt(1. + m[10]-m[0]-m[5])*0.5;
566  w = (m[1] - m[4])/(4.*z);
567  x = (m[2] + m[8])/(4.*z);
568  y = (m[6] + m[9])/(4.*z);
569  }
570  }
571  return *this;

574 template<typename T>
575 void Quat<T> :: toAxisAngle(T& aa, T& ax, T& ay, T& az) const {
576  T unit = w*w; // cos^2(theta/2)
577  if(unit < accuracyMin()){ // |cos x| must always be less than or equal to 1!
578  T invSinAngle = 1./sqrt(1. - unit); // = 1/sqrt(1 - cos^2(theta/2))
579 
580  aa = ::acos(w) * T(2);
581  ax = x * invSinAngle;
582  ay = y * invSinAngle;
583  az = z * invSinAngle;
584  } else {
585  aa = 0;
586  if(x == T(0) && y == T(0) && z == T(0)){
587  // axes are 0,0,0, change to a default:
588  ax = T(1);
589  ay = T(0);
590  az = T(0);
591  }
592  else{
593  // for small angles, axis is roughly equal to i,j,k components
594  // axes are close to zero, should be normalized:
595  Vec<3,T> v(x, y, z);
596  v.normalize();
597  ax = v[0];
598  ay = v[1];
599  az = v[2];
600  }
601  }
602 }

604 template<typename T>
605 inline void Quat<T> :: toEuler(T& az, T& el, T& ba) const {
606  /* Adapted from M. Baker:
607  http://www.euclideanspace.com/maths/geometry/rotations/conversions/quaternionToEuler/Quaternions.pdf
608  To make these equations work, 'e' had to be flipped in sign. */
609 
610  //T a=w, b=x, c=y, d=z, e= 1; // XYZ
611  //T a=w, b=x, c=z, d=y, e=-1; // XZY
612  //T a=w, b=y, c=z, d=x, e= 1; // YZX
613  T a=w, b=y, c=x, d=z, e=-1; // YXZ
614  //T a=w, b=z, c=x, d=y, e= 1; // ZXY
615  //T a=w, b=z, c=y, d=x, e=-1; // ZYX
616 
617  // Here we test if our second rotation will result in gimbal lock
618  T test = a*c + e * b*d;
619  if(test > 0.49999){ // singularity at north pole
620  az = atan2(b, a);
621  el = M_PI/2;
622  ba = 0;
623  return;
624  }
625  if(test < -0.49999){ // singularity at south pole
626  az = atan2(b, a);
627  el = -M_PI/2;
628  ba = 0;
629  return;
630  }
631 
632  T bsq = b*b;
633  T csq = c*c;
634  T dsq = d*d;
635 
636  az = ::atan2(2*(a*b - e * c*d), 1 - 2*(bsq + csq));
637  el = ::asin(2*test);
638  ba = ::atan2(2*(a*d - e * b*c), 1 - 2*(csq + dsq));
639 
640  // This version gives imprecise results:
641  // http://www.mathworks.com/access/helpdesk/help/toolbox/aeroblks/quaternionstoeulerangles.html

775  Quat<T> result;
776 
777  if (amt==T(0)) {
778  return input;
779  } else if (amt==T(1)) {
780  return target;
781  }
782 
783  int bflip = 0;
784  T dot_prod = input.dot(target);
785  T a, b;
786 
787  //clamp
788  dot_prod = (dot_prod < -1) ? -1 : ((dot_prod > 1) ? 1 : dot_prod);
789 
790  // if B is on opposite hemisphere from A, use -B instead
791  if (dot_prod < 0.0) {
792  dot_prod = -dot_prod;
793  bflip = 1;
794  }
795 
796  T cos_angle = acos(dot_prod);
797  if(Quat::abs(cos_angle) > eps()) {
798  T sine = sin(cos_angle);
799  T inv_sine = 1./sine;
800 
801  a = sin(cos_angle*(1.-amt)) * inv_sine;
802  b = sin(cos_angle*amt) * inv_sine;
803 
804  if (bflip) { b = -b; }
805  } else {
806  // nearly the same;
807  // approximate without trigonometry
808  a = amt;
809  b = 1.-amt;
810  }
811 
812  result.w = a*input.w + b*target.w;
813  result.x = a*input.x + b*target.x;
814  result.y = a*input.y + b*target.y;
815  result.z = a*input.z + b*target.z;
816 
817  result.normalize();
818  return result;

897 template<typename T>
898 Quat<T> Quat<T> :: getRotationTo(const Vec<3, T>& src, const Vec<3, T>& dst) {
899  // a . b = |a| |b| cos t
900  // Since |a| = |b| = 1, then
901  // a . b = cos t
902  T d = src.dot(dst);
903 
904  // vectors are the same
905  if (d >= 1.) {
906  return Quat<T>::identity();
907  }
908 
909  Quat<T> q;
910 
911  // vectors are nearly opposing
912  if (d < -0.999999999) {
913  // pick an axis to rotate around
914  Vec<3, T> axis = cross(Vec<3, T>(0, 1, 0), src);
915  // if colinear, pick another:
916  if (axis.magSqr() < 0.00000000001) {
917  axis = cross(Vec<3, T>(0, 0, 1), src);
918  }
919  //axis.normalize();
920  q.fromAxisAngle(M_PI, axis);
921  }
922  else {
923  /* Derive quaternion from a rotation axis and angle.
924 
925  The code used here is an optimization of fromAxisAngle:
926  T t2 = angle * T(0.5);
927  T sinft2 = sin(t2);
928  w = cos(t2);
929  x = ux * sinft2;
930  y = uy * sinft2;
931  z = uz * sinft2;
932  */
933  T s = sqrt((d+1.)*2);
934  T invs = 1./s;
935  Vec<3, T> c = cross(src, dst);
936  q.w = s * 0.5;
937  q.x = c[0] * invs;
938  q.y = c[1] * invs;
939  q.z = c[2] * invs;
940  }
941  return q.normalize();
942 }
