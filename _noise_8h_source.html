163 template<class T>
164 inline float NoisePink<T>::operator()(){
165  // phasor range: [1, 2048]
166  // [0, 10] trailing zeroes
167  ++mPhase;
168  if(mPhase != 2048){
169  uint32_t i = scl::trailingZeroes(mPhase); // # trailing zeroes is octave
170  float r = rnd::uniS_float(rng); // uniform random
171  mRunningSum += r - mOctave[i]; // add new & subtract stored
172  mOctave[i] = r; // store new
173  }
174  else{
175  mPhase = 0;
176  }
177 
178  // add white noise every sample
179  return (mRunningSum + rnd::uniS_float(rng)) * 0.083333333f;
