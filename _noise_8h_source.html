162 inline float NoisePink<T>::operator()(){
163  // phasor range: [1, 2048]
164  // [0, 10] trailing zeroes
165  ++mPhase;
166  if(mPhase != 2048){
167  uint32_t i = scl::trailingZeroes(mPhase); // # trailing zeroes is octave
168  float r = rnd::uniS_float(rng); // uniform random
169  mRunningSum += r - mOctave[i]; // add new & subtract stored

171  }
172  else{
173  mPhase = 0;

176  // add white noise every sample
