205  al_sec update() {
206  if (bUseRT) {
207  al_sec t2 = al_system_time() - mReferenceTime;
208  mDT = t2 - mNow;
209  mNow = t2;
210  mFrame++;
211  mFPS = mFPS + 0.1 * ((1./mDT) - mFPS);
212  } else {
213  mNow += mDT;
214  mFrame++;
215  mFPS = 1./mDT;
216  }
217  return now();
218  }
