561 template <int N, class T>
562 Vec<N,T> closestPointOnLineSegment(const Vec<N,T>& a, const Vec<N,T>& b, const Vec<N,T>& p){
563  auto ab = b - a;
564  auto dot = (p - a).dot(ab); // projection of ap onto ab
565  auto magAB = ab.magSqr();
566  auto frac = dot / magAB; // normalized distance along ab from a to the closest point
567 
568  // check if p projection is beyond endpoints of ab
569  if(frac < 0.) return a;
570  if(frac > 1.) return b;
571 
572  return a + ab * frac;
573 }
