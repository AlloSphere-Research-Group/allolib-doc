121 template <class Tf, class Tv>
122 inline Tv allpass(Tf f, const Tv& x, const Tv& y, Tv& o1){
123  // y[n] = a x[n] + x[n-1] - a y[n-1]
124  // = a (x[n] - y[n-1]) + x[n-1]
125 
126  //Tf a = f; // apx #1
127  //Tf a = Tf(0.4722)*f - Tf(0.2361); // apx #2
128 
129  //f = 1-f; // convert to sample delay
130  //f += Tf(0.618); // keeps 'a' near zero to dampen transients
131  //Tf a = (Tf(1)-f)/(Tf(1)+f);
132 
133  // Taylor approximation to above to avoid division
134  Tf a = Tf(0.5)*f - Tf(0.309); // = 0.5 - 0.5*(1-f + 0.618)
135  a = a*(Tf(1) + a*(Tf(1) + a*(Tf(1) + a)));
136 
137  return o1 = (y - o1) * a + x;
138 }

141 inline Tv cubic(Tf f, const Tv& w, const Tv& x, const Tv& y, const Tv& z){
142 // Tv c3 = (x - y)*(Tf)1.5 + (z - w)*(Tf)0.5;
143 // Tv c2 = w - x*(Tf)2.5 + y*(Tf)2. - z*(Tf)0.5;
144 // Tv c1 = (y - w)*(Tf)0.5;
145 // return ((c3 * f + c2) * f + c1) * f + x;
146 
147  // -w + 3x - 3y + z
148  // 2w - 5x + 4y - z
149  // c2 = w - 2x + y - c3
150 
151 // Tv c3 = (x - y)*(Tf)3 + z - w;
152 // Tv c2 = w - x*(Tf)2 + y - c3;
153 // Tv c1 = y - w;
154 // return (((c3 * f + c2) * f + c1)) * f * (Tf)0.5 + x;
155 
156 // Tv c3 = (x - y)*(Tf)1.5 + (z - w)*(Tf)0.5;
157 // Tv c2 = (y + w)*(Tf)0.5 - x - c3;
158 // Tv c1 = (y - w)*(Tf)0.5;
159 // return ((c3 * f + c2) * f + c1) * f + x;
160 
161  Tv c1 = (y - w)*Tf(0.5);
162  Tv c3 = (x - y)*Tf(1.5) + (z - w)*Tf(0.5);

173 template <class T> void lagrange(T * a, T delay, unsigned order){
174  for(unsigned i=0; i<=order; ++i){
175  T coef = T(1);
176  T i_f = T(i);
177  for(unsigned j=0; j<=order; ++j){
178  if(j != i){
179  T j_f = (T)j;
180  coef *= (delay - j_f) / (i_f - j_f);
181  }
182  }
183  *a++ = coef;
184  }
185 }
