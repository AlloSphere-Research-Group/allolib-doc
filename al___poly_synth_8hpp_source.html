935  inline void processVoices() {
936  if (mVoiceToInsertLock.try_lock()) {
937  if (mVoicesToInsert) {
938  // If lock acquired insert queued voices
939  if (mActiveVoices) {
940  auto voice = mVoicesToInsert;
941  while (voice->next) { // Find last voice to insert
942  voice = voice->next;
943  }
944  voice->next =
945  mActiveVoices; // Connect last inserted to previously active
946  mActiveVoices = mVoicesToInsert; // Put new voices in head
947  } else {
948  mActiveVoices = mVoicesToInsert;
949  }
950  if (verbose()) {
951  std::cout << "Voice on " << mVoicesToInsert->id() << std::endl;
952  }
953 
954  mVoicesToInsert = nullptr;
955  }
956  mVoiceToInsertLock.unlock();
957  }
958  if (mAllNotesOff) {
959  if (mFreeVoiceLock.try_lock()) {
960  mAllNotesOff = false;
961  if (mActiveVoices) {
962  auto voice = mActiveVoices->next;
963  SynthVoice *previousVoice = mActiveVoices;
964  previousVoice->id(-1);
965  while (voice) {
966  voice->id(-1);
967  previousVoice = voice;
968  voice = voice->next;
969  }
970  previousVoice->next =
971  mFreeVoices; // Connect last active voice to first free voice
972  mFreeVoices = mActiveVoices; // Move all voices to free voices
973  mActiveVoices = nullptr; // No active voices left
974  }
975  mFreeVoiceLock.unlock();
976  }
977  }

1017  inline void processInactiveVoices() {
1018  // Move inactive voices to free queue
1019  if (mFreeVoiceLock.try_lock()) { // Attempt to remove inactive voices
1020  // without waiting.
1021  auto *voice = mActiveVoices;
1022  SynthVoice *previousVoice = nullptr;
1023  while (voice) {
1024  if (!voice->active()) {
1025  int id = voice->id();
1026  // std::cout << " ****((((())))) **** Remove inactive voice:
1027  // " << voice << std::endl;
1028  if (previousVoice) {
1029  previousVoice->next = voice->next; // Remove from active list
1030  voice->next = mFreeVoices;
1031  mFreeVoices = voice; // Insert as head in free voices
1032  voice->id(-1); // Reset voice id
1033  voice->onFree();
1034  voice = previousVoice; // prepare next iteration
1035  } else { // Inactive is head of the list
1036  auto *nextVoice = voice->next;
1037  mActiveVoices = nextVoice; // Remove voice from list
1038  voice->next = mFreeVoices;
1039  mFreeVoices = voice; // Insert as head in free voices
1040  voice->id(-1); // Reset voice id
1041  voice->onFree();
1042  voice = voice->next; // prepare next iteration
1043  }
1044  for (auto cbNode : mFreeCallbacks) {
1045  cbNode.first(id, cbNode.second);
1046  }
1047  }
1048  previousVoice = voice;
1049  if (voice) {
1050  voice = voice->next;
1051  }
1052  }
1053  mFreeVoiceLock.unlock();
1054  }
