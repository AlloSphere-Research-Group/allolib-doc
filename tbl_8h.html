#define SYM_WIN (   period,
  phs0,
  eqn 
)
Value:
double inc = period / (double)(len);\
double phs = phs0;\
T * dst2 = dst + len - 1;\
*dst++ = T(eqn);\
LOOP(len>>1, 1){\
phs += inc;\
T val = T(eqn);\
*dst++ = val;\
*dst2-- = val;\
}
