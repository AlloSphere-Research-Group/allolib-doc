96  // intersect sphere
97  T intersectSphere( Vec<3,T> cen, T radius ){
98  Vec<3,T> o_c = o - cen;
99  T A = d.dot(d);
100  T B = 2. * ( d.dot( o_c ));
101  T C = (o_c.dot(o_c)) - radius*radius;
102  T det = B*B - 4*A*C;
103 
104  if( det > 0. ){
105  T t1 = (-B - sqrt(det) ) / (2.*A);
106  if ( t1 > 0. ) return t1;
107  T t2 = (-B + sqrt(det) ) / (2.*A);
108  if ( t2 > 0. ) return t2;
109 
110  } else if ( det == 0. ){
111  T t = -B / (2.*A);
112  if ( t > 0. ) return t;
113  }
114  return -1.; // will be ignoring negative intersections -1 qualifies a miss
115  }

121  T intersectBox(Vec<3,T> cen, Vec<3,T> scl){
122  // courtesy of http://www.cs.utah.edu/~awilliam/box/
123  float tmin, tmax, tymin, tymax, tzmin, tzmax;
124 
125  Vec<3,T> parameters[2];
126  Vec<3,T> min = cen - scl/2;
127  Vec<3,T> max = cen + scl/2;
128  parameters[0] = min;
129  parameters[1] = max;
130 
131  Vec<3,T> inv_direction = 1.0/d;
132  int sign[3];
133  sign[0] = (inv_direction.x < 0);
134  sign[1] = (inv_direction.y < 0);
135  sign[2] = (inv_direction.z < 0);
136 
137  tmin = (parameters[sign[0]].x - o.x) * inv_direction.x;
138  tmax = (parameters[1-sign[0]].x - o.x) * inv_direction.x;
139  tymin = (parameters[sign[1]].y - o.y) * inv_direction.y;
140  tymax = (parameters[1-sign[1]].y - o.y) * inv_direction.y;
141  if ( (tmin > tymax) || (tymin > tmax) )
142  return -1.0;
143  if (tymin > tmin)
144  tmin = tymin;
145  if (tymax < tmax)
146  tmax = tymax;
147  tzmin = (parameters[sign[2]].z - o.z) * inv_direction.z;
148  tzmax = (parameters[1-sign[2]].z - o.z) * inv_direction.z;
149  if ( (tmin > tzmax) || (tzmin > tmax) )
150  return -1.0;
151  if (tzmin > tmin)
152  tmin = tzmin;
153  if (tzmax < tmax)
154  tmax = tzmax;
155 
156  if(tmin < 0.0)
157  if(tmax < 0.0) return -1.0;
158  else return tmax;
159  else return tmin;
160  }
