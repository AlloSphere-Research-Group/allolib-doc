145  static Matrix4 rotate(float angle, const Vec<3, T> &v) {
146  Vec<3, T> axis(v);
147  axis.normalize();
148 
149  float c = cos(angle);
150  float s = sin(angle);
151 
152  Matrix4 m(
153  axis[0]*axis[0]*(1-c)+c,
154  axis[1]*axis[0]*(1-c)+axis[2]*s,
155  axis[0]*axis[2]*(1-c)-axis[1]*s,
156  0,
157 
158  axis[0]*axis[1]*(1-c)-axis[2]*s,
159  axis[1]*axis[1]*(1-c)+c,
160  axis[1]*axis[2]*(1-c)+axis[0]*s,
161  0,
162 
163  axis[0]*axis[2]*(1-c)+axis[1]*s,
164  axis[1]*axis[2]*(1-c)-axis[0]*s,
165  axis[2]*axis[2]*(1-c)+c,
166  0,
167 
168  0, 0, 0, 1
169  );
170 
171  return m;
172  }

207  static Matrix4 perspective(T l, T r, T b, T t, T n, T f) {
208  T W = r-l; T W2 = r+l;
209  T H = t-b; T H2 = t+b;

216  0, 0, -1, 0 );

225  static Matrix4 perspective(T fovy, T aspect, T near, T far) {
226  double f = 1./tan(fovy*M_DEG2RAD/2.);
227  T D = far-near; T D2 = far+near;

248  static Matrix4 perspective(
249  const Vec<3,T>& nearBL,
250  const Vec<3,T>& nearBR,
251  const Vec<3,T>& nearTL,
252  const Vec<3,T>& eye,
253  T near, T far)
254  {
255  Vec<3,T> va, vb, vc;
256  Vec<3,T> vr, vu, vn;
257  T l, r, b, t, d;
258 
259  // compute orthonormal basis for the screen
260  vr = (nearBR-nearBL).normalize(); // right vector
261  vu = (nearTL-nearBL).normalize(); // up vector
262  cross(vn, vr, vu); // normal(forward) vector (out from screen)
263  vn.normalize();
264 
265  // compute vectors from eye to screen corners:
266  va = nearBL-eye;
267  vb = nearBR-eye;
268  vc = nearTL-eye;
269 
270  // distance from eye to screen-plane
271  // = component of va along vector vn (normal to screen)
272  d = -va.dot(vn);
273 
274  // find extent of perpendicular projection
275  T nbyd = near/d;
276  l = vr.dot(va) * nbyd;
277  r = vr.dot(vb) * nbyd;
278  b = vu.dot(va) * nbyd; // not vd?
279  t = vu.dot(vc) * nbyd;
280 
281  return perspective(l, r, b, t, near, far);
282  }

295  static Matrix4 perspectiveOffAxis(T fovy, T aspect, T near, T far, T xShift, T focal) {
296  T top = near * tan(fovy*M_DEG2RAD*0.5); // height of view at distance = near
297  T bottom = -top;
298  T shift = -xShift*near/focal;
299  T left = -aspect*top + shift;
300  T right = aspect*top + shift;
301  return perspective(left, right, bottom, top, near, far);
302  }

313  static Matrix4 perspectiveOffAxis(T fovy, T aspect, T near, T far, T xShift, T yShift, T focal) {
314  double tanfovy = tan(fovy*M_DEG2RAD/2.);
315  T t = near * tanfovy; // height of view at distance = near
316  T b = -t;
317  T l = -aspect*t;
318  T r = aspect*t;
319 
320  T shift = -xShift*near/focal;
321  l += shift;
322  r += shift;
323  shift = -yShift*near/focal;
324  t += shift;
325  b += shift;
326 
327  return perspective(l, r, b, t, near, far);
328  }
