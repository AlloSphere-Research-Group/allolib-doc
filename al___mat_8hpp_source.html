645 template<int N, class T>
646 bool invert(Mat<N,T>& m){
647  // Get cofactor matrix, C
648  Mat<N,T> C = m.cofactorMatrix();
649 
650  // Compute determinant
651  T det = T(0);
652  for(int i=0; i<N; ++i){
653  det += m(0,i) * C(0,i);
654  }
655 
656  // Divide adjugate matrix, C^T, by determinant
657  if(det != T(0)){
658  m = (C.transpose() *= T(1)/det);
659  return true;
660  }
661 
662  return false;
663 }
