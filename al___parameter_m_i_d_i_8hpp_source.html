138  void connectNoteToValue(Parameter &param, int channel, float min, int low,
139  float max = -1, int high = -1) {
140  if (high == -1) {
141  max = min;
142  high = low;
143  }
144  for (int num = low; num <= high; ++num) {
145  NoteBinding newBinding;
146  newBinding.noteNumber = num;
147  if (num != high) {
148  newBinding.value = min + (max - min) * (num - low) / (high - low);
149  } else {
150  newBinding.value = max;
151  }
152  newBinding.channel = channel - 1;
153  newBinding.param = &param;
154  mNoteBindings.push_back(newBinding);
155  }
156  }

179  virtual void onMIDIMessage(const MIDIMessage &m) override {
180  if (m.type() & MIDIByte::CONTROL_CHANGE) {
181  for (ControlBinding binding : mControlBindings) {
182  if (m.channel() == binding.channel &&
183  m.controlNumber() == binding.controlNumber) {
184  float newValue =
185  binding.min + (m.controlValue() * (binding.max - binding.min));
186  binding.param->set(newValue);
187  }
188  }
189  }
190  if (m.type() & MIDIByte::NOTE_ON && m.velocity() > 0) {
191  for (NoteBinding binding : mNoteBindings) {
192  if (m.channel() == binding.channel &&
193  m.noteNumber() == binding.noteNumber) {
194  binding.param->set(binding.value);
195  }
196  }
197  for (IncrementBinding binding : mIncrementBindings) {
198  if (m.channel() == binding.channel &&
199  m.noteNumber() == binding.noteNumber) {
200  binding.param->set(binding.param->get() + binding.increment);
201  }
202  }
203  for (ToggleBinding binding : mToggleBindings) {
204  if (m.channel() == binding.channel &&
205  m.noteNumber() == binding.noteNumber) {
206  if (binding.toggle == true) {
207  binding.param->set(binding.param->get() == binding.param->max()
208  ? binding.param->min()
209  : binding.param->max());
210  } else {
211  binding.param->set(binding.param->max());
212  }
213  }
214  }
215  } else if (m.type() & MIDIByte::NOTE_OFF ||
216  (m.type() & MIDIByte::NOTE_ON && m.velocity() == 0)) {
217  for (ToggleBinding binding : mToggleBindings) {
218  if (m.channel() == binding.channel &&
219  m.noteNumber() == binding.noteNumber) {
220  if (binding.toggle != true) {
221  binding.param->set(binding.param->min());
222  }
223  }
224  }
225  }
226  if (mVerbose) {
227  m.print();
228  }
229  }
