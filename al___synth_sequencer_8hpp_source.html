542  std::list<SynthSequencerEvent> loadSequence(std::string sequenceName, double timeOffset = 0) {
543  std::unique_lock<std::mutex> lk(mLoadingLock);
544  std::list<SynthSequencerEvent> events;
545 // std::string fullName = buildFullPath(sequenceName);
546  std::string fullName = sequenceName;
547  std::ifstream f(fullName);
548  if (!f.is_open()) {
549  std::cout << "Could not open:" << fullName << std::endl;
550  return events;
551  }
552 
553  std::string line;
554  while (getline(f, line)) {
555  if (line.substr(0, 2) == "::") {
556  break;
557  }
558  std::stringstream ss(line);
559  int command = ss.get();
560  if (command == '@' && ss.get() == ' ') {
561  std::string name, start, durationText;
562  std::getline(ss, start, ' ');
563  std::getline(ss, durationText, ' ');
564  std::getline(ss, name, ' ');
565 
566  float startTime = std::stof(start);
567  double duration = std::stod(durationText);
568 
569  SynthVoice *newVoice = mPolySynth.getVoice(name);
570  if (newVoice) {
571  const int maxPFields = 32;
572  float pFields[maxPFields];
573 
574  int numFields = 0;
575  std::string field;
576  std::getline(ss, field, ' ');
577  while (field != "" && numFields < maxPFields) {
578  pFields[numFields] = std::stof(field);
579  numFields++;
580  std::getline(ss, field, ' ');
581  }
582  std::cout << "Pfields: ";
583  for (int i = 0; i < numFields; i++) {
584  std::cout << pFields[i] << " ";
585  }
586  std::cout << std::endl;
587 
588 
589  if (!newVoice->setParamFields(pFields, numFields)) {
590  std::cout << "Error setting pFields for voice of type " << name << ". Fields: ";
591  for (int i = 0; i < numFields; i++) {
592  std::cout << pFields[i] << " ";
593  }
594  std::cout << std::endl;
595  } else {
596  std::list<SynthSequencerEvent>::iterator insertedEvent;
597  double absoluteTime = timeOffset + startTime;
598  // Insert into event list, sorted.
599  auto position = events.begin();
600  while(position != events.end() && position->startTime < absoluteTime) {
601  position++;
602  }
603  insertedEvent = events.insert(position, SynthSequencerEvent());
604  // Add 0.1 padding to ensure all events play.
605  insertedEvent->startTime = absoluteTime;
606  insertedEvent->duration = duration;
607  insertedEvent->voice = newVoice;
608  std::cout << "Inserted event " << events.size() << " at time " << absoluteTime << std::endl;
609  }
610  }
611  std::cout << "Done reading sequence" << std::endl;
612  } else if (command == '+' && ss.get() == ' ') {
613  std::string name, idText, start;
614  std::getline(ss, start, ' ');
615  std::getline(ss, idText, ' ');
616  std::getline(ss, name, ' ');
617 
618  float startTime = std::stof(start);
619  int id = std::stoi(idText);
620 
621  SynthVoice *newVoice = mPolySynth.getVoice(name);
622  if (newVoice) {
623  newVoice->id(id);
624  const int maxPFields = 32;
625  float pFields[maxPFields];
626 
627  int numFields = 0;
628  std::string field;
629  std::getline(ss, field, ' ');
630  while (field != "" && numFields < maxPFields) {
631  pFields[numFields] = std::stof(field);
632  numFields++;
633  std::getline(ss, field, ' ');
634  }
635 // std::cout << "Pfields: ";
636 // for (int i = 0; i < numFields; i++) {
637 // std::cout << pFields[i] << " ";
638 // }
639 // std::cout << std::endl;
640  if (!newVoice->setParamFields(pFields, numFields)) {
641  std::cout << "Error setting pFields for voice of type " << name << ". Fields: ";
642  for (int i = 0; i < numFields; i++) {
643  std::cout << pFields[i] << " ";
644  }
645  std::cout << std::endl;
646  } else {
647  std::list<SynthSequencerEvent>::iterator insertedEvent;
648  double absoluteTime = timeOffset + startTime;
649  {
650 // // Insert into event list, sorted.
651  auto position = events.begin();
652  while(position != events.end() && position->startTime < absoluteTime) {
653  position++;
654  }
655  insertedEvent = events.insert(position, SynthSequencerEvent());
656  }
657  // Add 0.1 padding to ensure all events play.
658  insertedEvent->startTime = absoluteTime;
659  insertedEvent->duration = -1; // Turn on events have undetermined duration until a turn off is found later
660  insertedEvent->voice = newVoice;
661 // std::cout << "Inserted event " << id << " at time " << startTime << std::endl;
662  }
663  } else {
664  std::cout << "Warning: Unable to get free voice from PolySynth." << std::endl;
665  }
666  } else if (command == '-' && ss.get() == ' ') {
667  std::string time, idText;
668  std::getline(ss, time, ' ');
669  std::getline(ss, idText, ' ');
670  int id = std::stoi(idText);
671  double eventTime = std::stod(time);
672  for (SynthSequencerEvent &event: events) {
673  if (event.voice->id() == id && event.duration < 0) {
674  double duration = eventTime - event.startTime + timeOffset;
675  if (duration < 0) {
676  duration = 0;
677  }
678  event.duration = duration;
679 // std::cout << "Set event duration " << id << " to " << duration << std::endl;
680  break;
681  }
682  }
683  } else {
684  if (command > 0) {
685  std::cout << "Line ignored. Command: " << command << std::endl;
686  }
687  }
688  }
689  f.close();
690  if (f.bad()) {
691  std::cout << "Error reading:" << sequenceName << std::endl;
692  }
693  return events;
694  }

719 TSynthVoice *PolySynth::getVoice() {
720  std::unique_lock<std::mutex> lk(mFreeVoiceLock); // Only one getVoice() call at a time
721  SynthVoice *freeVoice = mFreeVoices;
722  SynthVoice *previousVoice = nullptr;
723  while (freeVoice) {
724  if (std::type_index(typeid(*freeVoice)) == std::type_index(typeid(TSynthVoice))) {
725  if (previousVoice) {
726  previousVoice->next = freeVoice->next;
727  } else {
728  mFreeVoices = freeVoice->next;
729  }
730  break;
731  }
732  previousVoice = freeVoice;
733  freeVoice = freeVoice->next;
734  }
735  if (!freeVoice) { // No free voice in list, so we need to allocate it
736  // TODO report current polyphony for more informed allocation of polyphony
737  std::cout << "Allocating voice of type " << typeid (TSynthVoice).name() << "." << std::endl;
738  freeVoice = new TSynthVoice;
739  }
740  return static_cast<TSynthVoice *>(freeVoice);
741 }

748  while (lastVoice->next) { lastVoice = lastVoice->next; }

752  }
