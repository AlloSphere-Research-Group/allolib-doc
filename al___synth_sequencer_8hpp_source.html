334 template <class TSynthVoice>
335 void SynthSequencer::addVoice(TSynthVoice *voice, double startTime,
336  double duration) {
337  std::list<SynthSequencerEvent>::iterator insertedEvent;
338  std::unique_lock<std::mutex> lk(mEventLock);
339  // Insert into event list, sorted.
340  auto position = mEvents.begin();
341  while (position != mEvents.end() && position->startTime < startTime) {
342  position++;
343  }
344  insertedEvent = mEvents.insert(position, SynthSequencerEvent());
345  insertedEvent->startTime = startTime;
346  insertedEvent->duration = duration;
347  insertedEvent->voice = voice;
348 }
