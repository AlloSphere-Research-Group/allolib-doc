578  std::list<SynthSequencerEvent> loadSequence(std::string sequenceName, double timeOffset = 0) {
579  std::unique_lock<std::mutex> lk(mLoadingLock);
580  std::list<SynthSequencerEvent> events;
581 // std::string fullName = buildFullPath(sequenceName);
582  std::string fullName = sequenceName;
583  std::ifstream f(fullName);
584  if (!f.is_open()) {
585  std::cout << "Could not open:" << fullName << std::endl;
586  return events;
587  }
588 
589  std::string line;
590  while (getline(f, line)) {
591  if (line.substr(0, 2) == "::") {
592  break;
593  }
594  std::stringstream ss(line);
595  int command = ss.get();
596  if (command == '@' && ss.get() == ' ') {
597  std::string name, start, durationText;
598  std::getline(ss, start, ' ');
599  std::getline(ss, durationText, ' ');
600  std::getline(ss, name, ' ');
601 
602  float startTime = std::stof(start);
603  double duration = std::stod(durationText);
604 
605  SynthVoice *newVoice = mPolySynth.getVoice(name);
606  if (newVoice) {
607  const int maxPFields = 32;
608  float pFields[maxPFields];
609 
610  int numFields = 0;
611  std::string field;
612  std::getline(ss, field, ' ');
613  while (field != "" && numFields < maxPFields) {
614  pFields[numFields] = std::stof(field);
615  numFields++;
616  std::getline(ss, field, ' ');
617  }
618  std::cout << "Pfields: ";
619  for (int i = 0; i < numFields; i++) {
620  std::cout << pFields[i] << " ";
621  }
622  std::cout << std::endl;
623 
624 
625  if (!newVoice->setParamFields(pFields, numFields)) {
626  std::cout << "Error setting pFields for voice of type " << name << ". Fields: ";
627  for (int i = 0; i < numFields; i++) {
628  std::cout << pFields[i] << " ";
629  }
630  std::cout << std::endl;
631  } else {
632  std::list<SynthSequencerEvent>::iterator insertedEvent;
633  double absoluteTime = timeOffset + startTime;
634  // Insert into event list, sorted.
635  auto position = events.begin();
636  while(position != events.end() && position->startTime < absoluteTime) {
637  position++;
638  }
639  insertedEvent = events.insert(position, SynthSequencerEvent());
640  // Add 0.1 padding to ensure all events play.
641  insertedEvent->startTime = absoluteTime;
642  insertedEvent->duration = duration;
643  insertedEvent->voice = newVoice;
644  std::cout << "Inserted event " << events.size() << " at time " << absoluteTime << std::endl;
645  }
646  }
647  std::cout << "Done reading sequence" << std::endl;
648  } else if (command == '+' && ss.get() == ' ') {
649  std::string name, idText, start;
650  std::getline(ss, start, ' ');
651  std::getline(ss, idText, ' ');
652  std::getline(ss, name, ' ');
653 
654  float startTime = std::stof(start);
655  int id = std::stoi(idText);
656 
657  SynthVoice *newVoice = mPolySynth.getVoice(name);
658  if (newVoice) {
659  newVoice->id(id);
660  const int maxPFields = 32;
661  float pFields[maxPFields];
662 
663  int numFields = 0;
664  std::string field;
665  std::getline(ss, field, ' ');
666  while (field != "" && numFields < maxPFields) {
667  pFields[numFields] = std::stof(field);
668  numFields++;
669  std::getline(ss, field, ' ');
670  }
671 // std::cout << "Pfields: ";
672 // for (int i = 0; i < numFields; i++) {
673 // std::cout << pFields[i] << " ";
674 // }
675 // std::cout << std::endl;
676  if (!newVoice->setParamFields(pFields, numFields)) {
677  std::cout << "Error setting pFields for voice of type " << name << ". Fields: ";
678  for (int i = 0; i < numFields; i++) {
679  std::cout << pFields[i] << " ";
680  }
681  std::cout << std::endl;
682  } else {
683  std::list<SynthSequencerEvent>::iterator insertedEvent;
684  double absoluteTime = timeOffset + startTime;
685  {
686 // // Insert into event list, sorted.
687  auto position = events.begin();
688  while(position != events.end() && position->startTime < absoluteTime) {
689  position++;
690  }
691  insertedEvent = events.insert(position, SynthSequencerEvent());
692  }
693  // Add 0.1 padding to ensure all events play.
694  insertedEvent->startTime = absoluteTime;
695  insertedEvent->duration = -1; // Turn on events have undetermined duration until a turn off is found later
696  insertedEvent->voice = newVoice;
697 // std::cout << "Inserted event " << id << " at time " << startTime << std::endl;
698  }
699  } else {
700  std::cout << "Warning: Unable to get free voice from PolySynth." << std::endl;
701  }
702  } else if (command == '-' && ss.get() == ' ') {
703  std::string time, idText;
704  std::getline(ss, time, ' ');
705  std::getline(ss, idText, ' ');
706  int id = std::stoi(idText);
707  double eventTime = std::stod(time);
708  for (SynthSequencerEvent &event: events) {
709  if (event.voice->id() == id && event.duration < 0) {
710  double duration = eventTime - event.startTime + timeOffset;
711  if (duration < 0) {
712  duration = 0;
713  }
714  event.duration = duration;
715 // std::cout << "Set event duration " << id << " to " << duration << std::endl;
716  break;
717  }
718  }
719  } else if (command == 't' && ss.get() == ' ') {
720  std::string time, idText;
721  std::getline(ss, time, ' ');
722  std::getline(ss, idText, ' ');
723  int id = std::stoi(idText);
724  double eventTime = std::stod(time);
725  for (SynthSequencerEvent &event: events) {
726  if (event.voice->id() == id && event.duration < 0) {
727  double duration = eventTime - event.startTime + timeOffset;
728  if (duration < 0) {
729  duration = 0;
730  }
731  event.duration = duration;
732 // std::cout << "Set event duration " << id << " to " << duration << std::endl;
733  break;
734  }
735  }
736  } else {
737  if (command > 0) {
738  std::cout << "Line ignored. Command: " << command << std::endl;
739  }
740  }
741  }
742  f.close();
743  if (f.bad()) {
744  std::cout << "Error reading:" << sequenceName << std::endl;
745  }
746  return events;
747  }

773  std::unique_lock<std::mutex> lk(mFreeVoiceLock); // Only one getVoice() call at a time
774  SynthVoice *freeVoice = mFreeVoices;
775  SynthVoice *previousVoice = nullptr;
776  while (freeVoice) {
777  if (std::type_index(typeid(*freeVoice)) == std::type_index(typeid(TSynthVoice))) {
778  if (previousVoice) {
779  previousVoice->next = freeVoice->next;
780  } else {
781  mFreeVoices = freeVoice->next;
782  }
783  break;
784  }
785  previousVoice = freeVoice;
786  freeVoice = freeVoice->next;
787  }
788  if (!freeVoice) { // No free voice in list, so we need to allocate it
789  // TODO report current polyphony for more informed allocation of polyphony
790  std::cout << "Allocating voice of type " << typeid (TSynthVoice).name() << "." << std::endl;
791  freeVoice = new TSynthVoice;
792  }
793  return static_cast<TSynthVoice *>(freeVoice);
