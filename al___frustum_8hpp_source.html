143  Vec<3,T> vmin = corner(0);
144  Vec<3,T> vmax = vmin;

146  for(int i=1; i<8; ++i){

185 template <class T>
186 int Frustum<T>::testBox(const Vec<3,T>& xyz, const Vec<3,T>& dim) const {
187  int result = INSIDE;
188  for(int i=0; i<6; ++i){
189  const Vec3d& plNrm = pl[i].normal();
190 
191 /*
192  The positive vertex is the vertex from the box that is further along
193  the normal's direction. The negative vertex is the opposite vertex.
194 
195  If the p-vertex is on the wrong side of the plane, the box can be
196  immediately rejected, as it falls completely outside the frustum. On the
197  other hand, if the p-vertex is on the right side of the plane, then
198  testing the whereabouts of the n-vertex tells if the box is totally on
199  the right side of the plane, or if the box intersects the plane.
200 */
201  // Is positive vertex outside?
202  Vec<3,T> vp = xyz;
203  if(plNrm[0] > 0) vp[0] += dim[0];
204  if(plNrm[1] > 0) vp[1] += dim[1];
205  if(plNrm[2] > 0) vp[2] += dim[2];
206  if(pl[i].inNegativeSpace(vp)) return OUTSIDE;
207 
208  // Is negative vertex outside?
209  Vec<3,T> vn = xyz;
210  if(plNrm[0] < 0) vn[0] += dim[0];
211  if(plNrm[1] < 0) vn[1] += dim[1];
212  if(plNrm[2] < 0) vn[2] += dim[2];
213  if(pl[i].inNegativeSpace(vn)) result = INTERSECT;
214  }
215  return result;
216 }
