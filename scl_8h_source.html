519 template<class T> T atan2Fast(T y, T x){
520 
521  T r, angle;
522  T ay = scl::abs(y) + T(1e-10); // kludge to prevent 0/0 condition
523 
524  if(x < T(0)){
525  r = (x + ay) / (ay - x);
526  angle = T(M_3PI_4);
527  }
528  else{
529  r = (x - ay) / (x + ay);
530  angle = T(M_PI_4);
531  }
532 
533  angle += (T(0.1963)*r*r - T(0.9817))*r;
534  return y < T(0) ? -angle : angle;
535 }
