412 template <class T>
413 Mesh& Mesh::transform(const Mat<4, T>& m, int begin, int end) {
414  if (end < 0)
415  end += vertices().size() + 1; // negative index wraps to end of array
416  for (int i = begin; i < end; ++i) {
417  Vertex& v = vertices()[i];
418  v.set(m * Vec<4, T>(v, 1));
419  }
420  return *this;
421 }
