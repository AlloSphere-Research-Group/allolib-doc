105  T attenuation(T dist) const {
106 
107  // No attenuation if below near distance
108  if(dist <= mNear) return T(1);
109 
110  switch(mLaw){
111  case ATTEN_LINEAR:
112  return dist < mFar ? T(1) - mScale*(dist - mNear) : mFarBias;
113 
114  case ATTEN_INVERSE:
115  return mNear / (mNear + mScale*(dist - mNear));
116 
117  case ATTEN_INVERSE_SQUARE:{
118  T nearSqr = mNear*mNear;
119  return nearSqr / (nearSqr + mScale*(dist*dist - nearSqr));
120  }
121 
122  default:
123  return T(1);
124  }
125  }
126 
127 protected:
128  T mNear, mFar; // clipping planes
129  T mFarBias; // bias on far clip (linear model only)
130  T mScale;
131  AttenuationLaw mLaw;
132 
133  DistAtten& setScale(){
134  switch(mLaw){
135  case ATTEN_LINEAR:
136  mScale = (T(1) - mFarBias)/(mFar - mNear);
137  break;
138 
139  // Note: For inverse laws, the attenuation factor at far clip is
140  // hard-coded. For INVERSE_SQUARE, it is the square of the value for
141  // INVERSE. This ensures a correct inverse power relationship.
142  case ATTEN_INVERSE:
143  mScale = (mNear/T(0.25) - mNear) / (mFar - mNear);
144  break;
145  case ATTEN_INVERSE_SQUARE:{
146  T nearSqr = mNear*mNear;
147  mScale = (nearSqr/T(0.25*0.25) - nearSqr) / (mFar*mFar - nearSqr);
148  }
149  break;
150  default:
151  mScale = 1;
152  }
153  return *this;
154  }
155 };
