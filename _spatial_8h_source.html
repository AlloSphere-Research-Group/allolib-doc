47  void gain(float v){

53  float g = gain();

74 template <typename T>
75 class Loop1P1Z{
76 public:
77 
79  void gain(float v){
80  mA0 = (mB1*0.5f + 0.5f)*v;
81  }
82 
84  void damping(float v){
85  float g = gain();
86  if(v>0.f) mB1 = 1.f-2.f*v;
87  gain(g);
88  }
89 
90  T operator()(T in){
91  mO1 = (in + mI1)*mA0 - mO1*mB1; // low-pass
92  //mO1 = (in - mI1)*mA0 - mO1*mB1; // high-pass
93  mI1 = in;
94  return mO1;
95  }

98  float gain() const {
99  return 2.f * mA0 / (mB1 + 1.f);
100  }
