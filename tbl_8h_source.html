509 template<class T>
510 void multiTriangle(T * dst, unsigned len, unsigned hrmLo, unsigned hrmHi){
511 
512  static const double triFactor = 8.0 / (M_PI * M_PI);
513  double inc = M_2PI / (double)len;
514  unsigned qLen = len >> 2;
515 
516  dst++;
517 
518  hrmLo |= 1; // next highest odd if even
519 
520  double factor = hrmLo & 0x2 ? -triFactor : triFactor;
521 
522  // Calculate first quadrant
523  for(unsigned i = hrmLo; i <= hrmHi; i+=2){
524 
525  double h = (double)i;
526  double phaseInc = h * inc;
527  double phs = phaseInc;
528  double amp = factor / (h * h);
529  factor = -factor;
530 
531  T * dst1 = dst;
532 
533  for(unsigned j=1; j<=qLen; ++j){
534  *dst1++ += T(amp * ::sin(phs));
535  phs += phaseInc;
536  }
537  }
538 
539  // Extrapolate remaining from [dbqp] symmetry
540  mem::reflectRight(dst, (len >> 1) - 1);
541  arr::mirror_dq(--dst, len);
542 }

727 template <int InvPower, class T>
728 void addSinesPow(
729  T * dst, unsigned len, int numh,
730  double hmul, double hshf, double amp, double hphs, double wphs, double cycles
731 ){
732  using std::pow;
733  const double inc1 = (M_2PI / len) * cycles;
734 
735  for(int j=0; j<numh; ++j){
736  const double h = j*hmul + hshf;
737  if(InvPower && 0==h) continue; // don't bother with DC component
738 
739  double A = amp;
740  switch(InvPower){
741  case 0: break;
742  case 1: A /= h; break;
743  case 2: A /= h*h; break;
744  case 3: A /= h*h*h; break;
745  default:A *= pow(h, -InvPower);
746  }
747 
748  const double P = (hphs + h*wphs) * M_2PI;
749  const double inch = inc1 * h;
750 
751  for(unsigned i=0; i<len; ++i){
752  dst[i] += A*getSin<T>(inch*i + P);
753  }
754  }
755 }
